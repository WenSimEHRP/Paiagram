43/// Direction in which a new node is created relatively to the parent node at which the split occurs.
44#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
45#[allow(missing_docs)]
46pub enum Split {
47    Left,
48    Right,
49    Above,
50    Below,
51}
52
53impl Split {
54    /// Returns whether the split is vertical.
55    pub const fn is_top_bottom(self) -> bool {
56        matches!(self, Split::Above | Split::Below)
57    }
58
59    /// Returns whether the split is horizontal.
60    pub const fn is_left_right(self) -> bool {
61        matches!(self, Split::Left | Split::Right)
62    }
63}

108/// Binary tree representing the relationships between [`Node`]s.
109///
110/// # Implementation details
111///
112/// The binary tree is stored in a [`Vec`] indexed by [`NodeIndex`].
113/// The root is always at index *0*.
114/// For a given node *n*:
115///  - left child of *n* will be at index *n * 2 + 1*.
116///  - right child of *n* will be at index *n * 2 + 2*.
117///
118/// For "Horizontal" nodes:
119///  - left child contains Left node.
120///  - right child contains Right node.
121///
122/// For "Vertical" nodes:
123///  - left child contains Top node.
124///  - right child contains Bottom node.
125#[derive(Clone)]
126#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize))]
127pub struct Tree<Tab> {
128    // Binary tree vector
129    pub(super) nodes: Vec<Node<Tab>>,
130    focused_node: Option<NodeIndex>,
131    // Whether all subnodes of the tree is collapsed
132    collapsed: bool,
133    collapsed_leaf_count: i32,
134}
135
136impl<Tab> fmt::Debug for Tree<Tab> {
137    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
138        f.debug_struct("Tree").finish_non_exhaustive()
139    }
140}

237    /// Counts and returns the number of tabs in the whole tree.
238    ///
239    /// # Examples
240    ///
241    /// ```rust
242    /// # use egui_dock::{DockState, NodeIndex, TabIndex};
243    /// let mut dock_state = DockState::new(vec!["node 1", "node 2", "node 3"]);
244    /// assert_eq!(dock_state.main_surface().num_tabs(), 3);

246    /// let [a, b] = dock_state.main_surface_mut().split_left(NodeIndex::root(), 0.5, vec!["tab 4", "tab 5"]);
247    /// assert_eq!(dock_state.main_surface().num_tabs(), 5);

254        let mut count = 0;
255        for node in self.nodes.iter() {
256            if let Node::Leaf(leaf) = node {
257                count += leaf.tabs.len();
258            }
259        }
260        count

439    /// Creates two new nodes by splitting a given `parent` node and assigns them as its children. The first (old) node
440    /// inherits content of the `parent` from before the split, and the second (new) uses `new`.
441    ///
442    /// `fraction` (in range 0..=1) specifies how much of the `parent` node's area the old node will attempt to occupy
443    /// after the split.
444    ///
445    /// The new node is placed relatively to the old node, in the direction specified by `split`.
446    ///
447    /// Returns the indices of the old node and the new node.
448    ///
449    /// # Panics
450    ///
451    /// If `fraction` isn't in range 0..=1.
452    ///
453    /// If `new` is an [`Empty`](Node::Empty), [`Horizontal`](Node::Horizontal) or [`Vertical`](Node::Vertical) node.
454    ///
455    /// If `new` is a [`Leaf`](Node::Leaf) node without any tabs.
456    ///
457    /// If `parent` points to an [`Empty`](Node::Empty) node.
458    ///
459    /// # Example
460    ///
461    /// ```rust
462    /// # use egui_dock::{DockState, SurfaceIndex, NodeIndex, Split, Node};
463    /// let mut dock_state = DockState::new(vec!["tab 1", "tab 2"]);
464    ///
465    /// // At this point, the main surface only contains the leaf with tab 1 and 2.
466    /// assert!(dock_state.main_surface().root_node().unwrap().is_leaf());
467    ///
468    /// // Splits the node, giving 50% of the space to the new nodes and 50% to the old ones.
469    /// let [old, new] = dock_state.main_surface_mut()
470    ///     .split(NodeIndex::root(), Split::Below, 0.5, Node::leaf_with(vec!["tab 3"]));
471    ///
472    /// assert!(dock_state.main_surface().root_node().unwrap().is_parent());
473    /// assert!(dock_state[SurfaceIndex::main()][old].is_leaf());
474    /// assert!(dock_state[SurfaceIndex::main()][new].is_leaf());
475    /// ```
476    pub fn split(
477        &mut self,
478        parent: NodeIndex,
479        split: Split,
480        fraction: f32,
481        new: Node<Tab>,
482    ) -> [NodeIndex; 2] {
483        let old = self[parent].split(split, fraction);
484        assert!(old.is_leaf() || old.is_parent());
485        assert_ne!(new.tabs_count(), 0);
486        // Resize vector to fit the new size of the binary tree.
487        {
488            let index = self.nodes.iter().rposition(|n| !n.is_empty()).unwrap_or(0);
489            let level = NodeIndex(index).level();
490            self.nodes
491                .resize_with((1 << (level + 1)) - 1, || Node::Empty);
492        }
493
494        let index = match split {
495            Split::Left | Split::Above => [parent.right(), parent.left()],
496            Split::Right | Split::Below => [parent.left(), parent.right()],
497        };
498
499        // If the node were splitting is a parent, all it's children need to be moved.
500        if old.is_parent() {
501            let levels_to_move = NodeIndex(self.nodes.len()).level() - index[0].level();
502
503            // Level 0 is ourself, which is done when we assign self[index[0]] = old, so start at 1.
504            for level in (1..levels_to_move).rev() {
505                // Old child indices for this level
506                let old_start = parent.children_at(level).start;
507                // New child indices for this level
508                let new_start = index[0].children_at(level).start;
509
510                // Children to be moved this level change
511                let len = 1 << level;
512
513                // Swap self[old_start..(old_start+len)] with self[new_start..(new_start+len)]
514                // (the new part will only contain empty entries).
515                let (old_range, new_range) = {
516                    let (first_part, second_part) = self.nodes.split_at_mut(new_start);
517                    // Cut to length.
518                    (
519                        &mut first_part[old_start..old_start + len],
520                        &mut second_part[..len],
521                    )
522                };
523                old_range.swap_with_slice(new_range);
524            }
525        }
526
527        self[index[0]] = old;
528        self[index[1]] = new;
529
530        self.focused_node = Some(index[1]);
531        self.node_update_collapsed(index[1]);
532
533        index
534    }

536    fn first_leaf(&self, top: NodeIndex) -> Option<NodeIndex> {
537        let left = top.left();
538        let right = top.right();
539        match (self.nodes.get(left.0), self.nodes.get(right.0)) {
540            (Some(&Node::Leaf { .. }), _) => Some(left),
541            (_, Some(&Node::Leaf { .. })) => Some(right),
542
543            (
544                Some(Node::Horizontal { .. } | Node::Vertical { .. }),
545                Some(Node::Horizontal { .. } | Node::Vertical { .. }),
546            ) => self.first_leaf(left).or(self.first_leaf(right)),
547            (Some(Node::Horizontal { .. } | Node::Vertical { .. }), _) => self.first_leaf(left),
548            (_, Some(Node::Horizontal { .. } | Node::Vertical { .. })) => self.first_leaf(right),
549
550            (None, None)
551            | (Some(&Node::Empty), None)
552            | (None, Some(&Node::Empty))
553            | (Some(&Node::Empty), Some(&Node::Empty)) => None,
554        }
555    }

584    /// Removes the given node from the [`Tree`].
585    ///
586    /// # Panics
587    ///
588    /// - If the tree is empty.
589    /// - If the node at index `node` is not a [`Leaf`](Node::Leaf).
590    pub fn remove_leaf(&mut self, node: NodeIndex) {
591        assert!(!self.is_empty());
592        assert!(self[node].is_leaf());
593
594        let Some(parent) = node.parent() else {
595            self.nodes.clear();
596            return;
597        };
598
599        if Some(node) == self.focused_node {
600            self.focused_node = None;
601            let mut node = node;
602            while let Some(parent) = node.parent() {
603                let next = if node.is_left() {
604                    parent.right()
605                } else {
606                    parent.left()
607                };
608                if self.nodes.get(next.0).is_some_and(|node| node.is_leaf()) {
609                    self.focused_node = Some(next);
610                    break;
611                }
612                if let Some(node) = self.first_leaf(next) {
613                    self.focused_node = Some(node);
614                    break;
615                }
616                node = parent;
617            }
618        }
619
620        self[parent] = Node::Empty;
621        self[node] = Node::Empty;
622
623        let mut level = 0;
624
625        if node.is_left() {
626            'left_end: loop {
627                let dst = parent.children_at(level);
628                let src = parent.children_right(level + 1);
629                for (dst, src) in dst.zip(src) {
630                    if src >= self.nodes.len() {
631                        break 'left_end;
632                    }
633                    if Some(NodeIndex(src)) == self.focused_node {
634                        self.focused_node = Some(NodeIndex(dst));
635                    }
636                    self.nodes[dst] = std::mem::replace(&mut self.nodes[src], Node::Empty);
637                }
638                level += 1;
639            }
640        } else {
641            'right_end: loop {
642                let dst = parent.children_at(level);
643                let src = parent.children_left(level + 1);
644                for (dst, src) in dst.zip(src) {
645                    if src >= self.nodes.len() {
646                        break 'right_end;
647                    }
648                    if Some(NodeIndex(src)) == self.focused_node {
649                        self.focused_node = Some(NodeIndex(dst));
650                    }
651                    self.nodes[dst] = std::mem::replace(&mut self.nodes[src], Node::Empty);
652                }
653                level += 1;
654            }
655        }
656        // Ensure that there are no trailing `Node::Empty` items
657        while let Some(last_index) = self.nodes.len().checked_sub(1).map(NodeIndex) {
658            if self[last_index].is_empty()
659                && last_index.parent().is_some_and(|pi| !self[pi].is_parent())
660            {
661                self.nodes.pop();
662            } else {
663                break;
664            }
665        }
666    }

668    /// Pushes a tab to the first `Leaf` it finds or create a new leaf if an `Empty` node is encountered.
669    pub fn push_to_first_leaf(&mut self, tab: Tab) {
670        for (index, node) in &mut self.nodes.iter_mut().enumerate() {
671            match node {
672                Node::Leaf(leaf) => {
673                    leaf.active = TabIndex(leaf.tabs.len());
674                    leaf.tabs.push(tab);
675                    self.focused_node = Some(NodeIndex(index));
676                    return;
677                }
678                Node::Empty => {
679                    *node = Node::leaf(tab);
680                    self.focused_node = Some(NodeIndex(index));
681                    return;
682                }
683                _ => {}
684            }
685        }
686        assert!(self.nodes.is_empty());
687        self.nodes.push(Node::leaf_with(vec![tab]));
688        self.focused_node = Some(NodeIndex(0));
689    }

691    /// Sets which is the active tab within a specific node.
692    #[inline]
693    pub fn set_active_tab(
694        &mut self,
695        node_index: impl Into<NodeIndex>,
696        tab_index: impl Into<TabIndex>,
697    ) {
698        if let Some(Node::Leaf(leaf)) = self.nodes.get_mut(node_index.into().0) {
699            leaf.set_active_tab(tab_index);
700        };
701    }

703    /// Pushes `tab` to the currently focused leaf.
704    ///
705    /// If no leaf is focused it will be pushed to the first available leaf.
706    ///
707    /// If no leaf is available then a new leaf will be created.
708    pub fn push_to_focused_leaf(&mut self, tab: Tab) {
709        match self.focused_node {
710            Some(node) => {
711                if self.nodes.is_empty() {
712                    self.nodes.push(Node::leaf(tab));
713                    self.focused_node = Some(NodeIndex::root());
714                } else {
715                    match &mut self[node] {
716                        Node::Empty => {
717                            self[node] = Node::leaf(tab);
718                            self.focused_node = Some(node);
719                        }
720                        Node::Leaf(leaf) => {
721                            leaf.append_tab(tab);
722                            self.focused_node = Some(node);
723                        }
724                        _ => {
725                            self.push_to_first_leaf(tab);
726                        }
727                    }
728                }
729            }
730            None => {
731                if self.nodes.is_empty() {
732                    self.nodes.push(Node::leaf(tab));
733                    self.focused_node = Some(NodeIndex::root());
734                } else {
735                    self.push_to_first_leaf(tab);
736                }
737            }
738        }
739    }

741    /// Removes the tab at the given ([`NodeIndex`], [`TabIndex`]) pair.
742    ///
743    /// If the node is emptied after the tab is removed, the node will also be removed.
744    ///
745    /// Returns the removed tab if it exists, or `None` otherwise.
746    pub fn remove_tab(&mut self, (node_index, tab_index): (NodeIndex, TabIndex)) -> Option<Tab> {
747        let node = &mut self[node_index];
748        let tab = node.remove_tab(tab_index);
749        if node.tabs_count() == 0 {
750            self.remove_leaf(node_index);
751        }
752        tab
753    }

755    /// Returns a new [`Tree`] while mapping and filtering the tab type.
756    /// Any remaining empty [`Node`]s are removed.
757    pub fn filter_map_tabs<F, NewTab>(&self, mut function: F) -> Tree<NewTab>
758    where
759        F: FnMut(&Tab) -> Option<NewTab>,
760    {
761        let Tree {
762            focused_node,
763            nodes,
764            collapsed,
765            collapsed_leaf_count,
766        } = self;
767        let mut emptied_nodes = HashSet::default();
768        let nodes = nodes
769            .iter()
770            .enumerate()
771            .map(|(index, node)| {
772                let filtered_node = node.filter_map_tabs(&mut function);
773                if filtered_node.is_empty() && !node.is_empty() {
774                    emptied_nodes.insert(NodeIndex(index));
775                }
776                filtered_node
777            })
778            .collect();
779        let mut new_tree = Tree {
780            nodes,
781            focused_node: *focused_node,
782            collapsed: *collapsed,
783            collapsed_leaf_count: *collapsed_leaf_count,
784        };
785        new_tree.balance(emptied_nodes);
786        new_tree
787    }

789    /// Returns a new [`Tree`] while mapping the tab type.
790    pub fn map_tabs<F, NewTab>(&self, mut function: F) -> Tree<NewTab>
791    where
792        F: FnMut(&Tab) -> NewTab,
793    {
794        self.filter_map_tabs(move |tab| Some(function(tab)))
795    }

797    /// Returns a new [`Tree`] while filtering the tab type.
798    /// Any remaining empty [`Node`]s are removed.
799    pub fn filter_tabs<F>(&self, mut predicate: F) -> Tree<Tab>
800    where
801        F: FnMut(&Tab) -> bool,
802        Tab: Clone,
803    {
804        self.filter_map_tabs(move |tab| predicate(tab).then(|| tab.clone()))
805    }

807    /// Removes all tabs for which `predicate` returns `false`.
808    /// Any remaining empty [`Node`]s are also removed.
809    pub fn retain_tabs<F>(&mut self, mut predicate: F)
810    where
811        F: FnMut(&mut Tab) -> bool,
812    {
813        let mut emptied_nodes = HashSet::default();
814        for (index, node) in self.nodes.iter_mut().enumerate() {
815            node.retain_tabs(&mut predicate);
816            if node.is_empty() {
817                emptied_nodes.insert(NodeIndex(index));
818            }
819        }
820        self.balance(emptied_nodes);
821    }

843    fn balance(&mut self, emptied_nodes: HashSet<NodeIndex>) {
844        let mut emptied_parents = HashSet::default();
845        for parent_index in emptied_nodes.into_iter().filter_map(|ni| ni.parent()) {
846            if !self[parent_index].is_parent() {
847                continue;
848            } else if self[parent_index.left()].is_empty() && self[parent_index.right()].is_empty()
849            {
850                self[parent_index] = Node::Empty;
851                emptied_parents.insert(parent_index);
852            } else if self[parent_index.left()].is_empty() {
853                self.nodes.swap(parent_index.0, parent_index.right().0);
854                self[parent_index.right()] = Node::Empty;
855            } else if self[parent_index.right()].is_empty() {
856                self.nodes.swap(parent_index.0, parent_index.left().0);
857                self[parent_index.left()] = Node::Empty;
858            }
859        }
860        if !emptied_parents.is_empty() {
861            self.balance(emptied_parents);
862        }
863    }

865    /// Updates the collapsed state of the node and its parents.
866    pub(crate) fn node_update_collapsed(&mut self, node_index: NodeIndex) {
867        let collapsed = self[node_index].is_collapsed();
868        if !collapsed {
869            // Recursively notify parent nodes that the leaf has expanded
870            let mut parent_index_option = node_index.parent();
871            while let Some(parent_index) = parent_index_option {
872                parent_index_option = parent_index.parent();
873
874                // Update collapsed leaf count and collapse status
875                let left_count = self[parent_index.left()].collapsed_leaf_count();
876                let right_count = self[parent_index.right()].collapsed_leaf_count();
877                self[parent_index].set_collapsed(false);
878
879                if self[parent_index].is_horizontal() {
880                    self[parent_index].set_collapsed_leaf_count(max(left_count, right_count));
881                } else {
882                    self[parent_index].set_collapsed_leaf_count(left_count + right_count);
883                }
884            }
885            self.set_collapsed(false);
886            let root_index = NodeIndex::root();
887            self.set_collapsed_leaf_count(self[root_index].collapsed_leaf_count());
888        } else {
889            // Recursively notify parent nodes that the leaf has collapsed
890            let mut parent_index_option = node_index.parent();
891            while let Some(parent_index) = parent_index_option {
892                parent_index_option = parent_index.parent();
893
894                // Update collapsed leaf count and collapse status
895                let left_count = self[parent_index.left()].collapsed_leaf_count();
896                let right_count = self[parent_index.right()].collapsed_leaf_count();
897
898                if self[parent_index].is_horizontal() {
899                    self[parent_index].set_collapsed_leaf_count(max(left_count, right_count));
900                } else {
901                    self[parent_index].set_collapsed_leaf_count(left_count + right_count);
902                }
903
904                if self[parent_index.left()].is_collapsed()
905                    && self[parent_index.right()].is_collapsed()
906                {
907                    self[parent_index].set_collapsed(true);
908                }
909            }
910            if self.root_node().is_some_and(|root| root.is_collapsed()) {
911                self.set_collapsed(true);
912                let root_index = NodeIndex::root();
913                self.set_collapsed_leaf_count(self[root_index].collapsed_leaf_count());
914            }
915        }
916    }

918    /// Find a given tab based on ``predicate``.
919    ///
920    /// Returns the indices in where that node and tab is in this surface.
921    ///
922    /// The returned [`NodeIndex`] will always point to a [`Node::Leaf`].
923    ///
924    /// In case there are several hits, only the first is returned.
925    pub fn find_tab_from(&self, predicate: impl Fn(&Tab) -> bool) -> Option<(NodeIndex, TabIndex)> {
926        for (node_index, node) in self.nodes.iter().enumerate() {
927            if let Some(tabs) = node.tabs() {
928                for (tab_index, tab) in tabs.iter().enumerate() {
929                    if predicate(tab) {
930                        return Some((node_index.into(), tab_index.into()));
931                    }
932                }
933            };
934        }
935        None
936    }

939impl<Tab> Tree<Tab>
940where
941    Tab: PartialEq,
942{
943    /// Find the given tab.
944    ///
945    /// Returns in which node and where in that node the tab is.
946    ///
947    /// The returned [`NodeIndex`] will always point to a [`Node::Leaf`].
948    ///
949    /// In case there are several hits, only the first is returned.
950    pub fn find_tab(&self, needle_tab: &Tab) -> Option<(NodeIndex, TabIndex)> {
951        self.find_tab_from(|tab| tab == needle_tab)
952    }
