9/// Represents an abstract node of a [`Tree`](crate::Tree).
10#[derive(Clone, Debug)]
11#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize))]
12pub enum Node<Tab> {
13    /// Empty node.
14    Empty,
15
16    /// Contains the actual tabs.
17    Leaf(LeafNode<Tab>),
18
19    /// Parent node in the vertical orientation.
20    Vertical(SplitNode),
21
22    /// Parent node in the horizontal orientation.
23    Horizontal(SplitNode),
24}
25
26impl<Tab> Node<Tab> {
27    /// Constructs a leaf node with a given `tab`.
28    #[inline(always)]
29    pub fn leaf(tab: Tab) -> Self {
30        Self::Leaf(LeafNode::new(vec![tab]))
31    }
32
33    /// Get immutable access to the leaf data of this node, if it contains any (i.e is a leaf)
34    pub fn get_leaf(&self) -> Option<&LeafNode<Tab>> {
35        match self {
36            Node::Leaf(leaf_node) => Some(leaf_node),
37            _ => None,
38        }
39    }

41    /// Get mutable access to the leaf data of this node, if it contains any (i.e is a leaf)
42    pub fn get_leaf_mut(&mut self) -> Option<&mut LeafNode<Tab>> {
43        match self {
44            Node::Leaf(leaf_node) => Some(leaf_node),
45            _ => None,
46        }
47    }

49    /// Constructs a leaf node with a given list of `tabs`.
50    #[inline(always)]
51    pub const fn leaf_with(tabs: Vec<Tab>) -> Self {
52        Self::Leaf(LeafNode::new(tabs))
53    }

55    /// Sets the area occupied by the node.
56    ///
57    /// If the node is a ``Node::Empty``, this will do nothing.
58    #[inline]
59    pub fn set_rect(&mut self, new_rect: Rect) {
60        match self {
61            Self::Empty => (),
62            Self::Leaf(leaf) => leaf.set_rect(new_rect),
63            Self::Vertical(split) | Self::Horizontal(split) => split.set_rect(new_rect),
64        }
65    }
66
67    /// Get a [`Rect`] occupied by the node, could be used e.g. to draw a highlight rect around a node.
68    ///
69    /// Returns [`None`] if node is of the [`Empty`](Node::Empty) variant.
70    #[inline]
71    pub fn rect(&self) -> Option<Rect> {
72        match self {
73            Self::Empty => None,
74            Self::Leaf(leaf) => Some(leaf.rect()),
75            Self::Vertical(split) | Self::Horizontal(split) => Some(split.rect()),
76        }
77    }

79    /// Returns `true` if the node is a [`Empty`](Node::Empty), otherwise `false`.
80    #[inline(always)]
81    pub const fn is_empty(&self) -> bool {
82        matches!(self, Self::Empty)
83    }

85    /// Returns `true` if the node is a [`Leaf`](Node::Leaf), otherwise `false`.
86    #[inline(always)]
87    pub const fn is_leaf(&self) -> bool {
88        matches!(self, Self::Leaf { .. })
89    }
90
91    /// Returns `true` if the node is a [`Horizontal`](Node::Horizontal), otherwise `false`.
92    #[inline(always)]
93    pub const fn is_horizontal(&self) -> bool {
94        matches!(self, Self::Horizontal { .. })
95    }
96
97    /// Returns `true` if the node is a [`Vertical`](Node::Vertical), otherwise `false`.
98    #[inline(always)]
99    pub const fn is_vertical(&self) -> bool {
100        matches!(self, Self::Vertical { .. })
101    }
102
103    /// Returns `true` if the node is either [`Horizontal`](Node::Horizontal) or [`Vertical`](Node::Vertical),
104    /// otherwise `false`.
105    #[inline(always)]
106    pub const fn is_parent(&self) -> bool {
107        self.is_horizontal() || self.is_vertical()
108    }
109
110    /// Returns `true` if the node is collapsed, otherwise `false`.
111    #[inline(always)]
112    pub fn is_collapsed(&self) -> bool {
113        match self {
114            Node::Leaf(leaf) => leaf.collapsed,
115            Node::Horizontal(split) | Node::Vertical(split) => split.fully_collapsed,
116            Node::Empty => false,
117        }
118    }
119
120    /// Returns the number of layers of collapsed leaf subnodes.
121    pub fn collapsed_leaf_count(&self) -> i32 {
122        match self {
123            Node::Horizontal(split) | Node::Vertical(split) => split.collapsed_leaf_count,
124            Node::Leaf(leaf) => {
125                if leaf.collapsed {
126                    1
127                } else {
128                    0
129                }
130            }
131            Node::Empty => 0,
132        }
133    }

135    /// Replaces the node with [`Horizontal`](Node::Horizontal) or [`Vertical`](Node::Vertical) (depending on `split`)
136    /// and assigns an empty rect to it.
137    ///
138    /// # Panics
139    ///
140    /// If `fraction` isn't in range 0..=1.
141    #[inline]
142    pub fn split(&mut self, split: Split, fraction: f32) -> Self {
143        assert!((0.0..=1.0).contains(&fraction));
144        let rect = Rect::NOTHING;
145        let src = match split {
146            Split::Left | Split::Right => Node::Horizontal(SplitNode::new(
147                rect,
148                fraction,
149                self.is_collapsed(),
150                self.collapsed_leaf_count(),
151            )),
152            Split::Above | Split::Below => Node::Vertical(SplitNode::new(
153                rect,
154                fraction,
155                self.is_collapsed(),
156                self.collapsed_leaf_count(),
157            )),
158        };
159        std::mem::replace(self, src)
160    }

234    ///
235    /// # Panics
236    ///

238    ///
239    /// If `self` is not a [`Leaf`](Node::Leaf) node.
240    ///
241    /// # Examples
242    ///
243    /// ```rust

246    /// assert_eq!(dock_state.main_surface().root_node().unwrap().tabs_count(), 1);
247    ///
248    /// dock_state.main_surface_mut().root_node_mut().unwrap().append_tab("another tab");
249    /// assert_eq!(dock_state.main_surface().root_node().unwrap().tabs_count(), 2);

254        match self {

256            _ => panic!("node was not a leaf"),
257        }

262    /// # Panics

324    /// Returns a new [`Node`] while mapping and filtering the tab type.
325    /// If this [`Node`] remains empty, it will change to [`Node::Empty`].
326    pub fn filter_map_tabs<F, NewTab>(&self, function: F) -> Node<NewTab>
327    where
328        F: FnMut(&Tab) -> Option<NewTab>,
329    {
330        match self {
331            Node::Leaf(leaf) => {
332                let LeafNode {
333                    rect,
334                    viewport,
335                    tabs,
336                    active,
337                    scroll,
338                    collapsed,
339                } = leaf;
340                let tabs: Vec<_> = tabs.iter().filter_map(function).collect();
341                if tabs.is_empty() {
342                    Node::Empty
343                } else {
344                    Node::Leaf(LeafNode {
345                        rect: *rect,
346                        viewport: *viewport,
347                        tabs,
348                        active: *active,
349                        scroll: *scroll,
350                        collapsed: *collapsed,
351                    })
352                }
353            }
354            Node::Empty => Node::Empty,
355            Node::Vertical(split) => Node::Vertical(split.clone()),
356            Node::Horizontal(split) => Node::Horizontal(split.clone()),
357        }
358    }
359
360    /// Returns a new [`Node`] while mapping the tab type.
361    pub fn map_tabs<F, NewTab>(&self, mut function: F) -> Node<NewTab>
362    where
363        F: FnMut(&Tab) -> NewTab,
364    {
365        self.filter_map_tabs(move |tab| Some(function(tab)))
366    }
367
368    /// Returns a new [`Node`] while filtering the tab type.
369    /// If this [`Node`] remains empty, it will change to [`Node::Empty`].
370    pub fn filter_tabs<F>(&self, mut predicate: F) -> Node<Tab>
371    where
372        F: FnMut(&Tab) -> bool,
373        Tab: Clone,
374    {
375        self.filter_map_tabs(move |tab| predicate(tab).then(|| tab.clone()))
376    }
377
378    /// Removes all tabs for which `predicate` returns `false`.
379    /// If this [`Node`] remains empty, it will change to [`Node::Empty`].
380    pub fn retain_tabs<F>(&mut self, predicate: F)
381    where
382        F: FnMut(&mut Tab) -> bool,
383    {
384        if let Node::Leaf(leaf) = self {
385            leaf.retain_tabs(predicate);
386            if leaf.tabs.is_empty() {
387                *self = Node::Empty;
388            }
389        }
390    }
