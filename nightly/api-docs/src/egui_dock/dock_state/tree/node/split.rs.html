3///the inner data of a [``Node::Horizontal``](crate::Node)/[``Node::Vertical``](crate::Node), which splits into two further nodes.
4#[derive(Clone, Debug)]
5#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize))]
6pub struct SplitNode {
7    /// The rectangle in which all children of this node are drawn.
8    pub rect: Rect,
9
10    /// The fraction taken by the top child of this node.
11    pub fraction: f32,
12
13    /// Whether all subnodes are collapsed.
14    pub fully_collapsed: bool,
15
16    /// The number of collapsed leaf subnodes.
17    pub collapsed_leaf_count: i32,
18}
19
20impl SplitNode {
21    /// Create a new ``SplitNode``
22    pub const fn new(
23        rect: Rect,
24        fraction: f32,
25        fully_collapsed: bool,
26        collapsed_leaf_count: i32,
27    ) -> Self {
28        Self {
29            rect,
30            fraction,
31            fully_collapsed,
32            collapsed_leaf_count,
33        }
34    }
35    /// Set the Area which this ``SplitNode`` occupies.
36    #[inline]
37    pub fn set_rect(&mut self, new_rect: Rect) {
38        self.rect = new_rect;
39    }
40
41    /// Get the Area which this ``SplitNode`` occupies.
42    pub fn rect(&self) -> Rect {
43        self.rect
