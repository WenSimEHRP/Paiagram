32    Surface(SurfaceIndex),
33    Node(SurfaceIndex, NodeIndex),
34    Tab(SurfaceIndex, NodeIndex, TabIndex),

37impl TreeComponent {
38    pub(super) fn as_tab_destination(&self) -> TabDestination {
39        match *self {
40            TreeComponent::Surface(surface) => TabDestination::EmptySurface(surface),
41            TreeComponent::Node(dst_surf, dst_node) => {
42                TabDestination::Node(dst_surf, dst_node, TabInsert::Append)
43            }
44            TreeComponent::Tab(dst_surf, dst_node, tab_index) => {
45                TabDestination::Node(dst_surf, dst_node, TabInsert::Insert(tab_index))
46            }
47        }
48    }
49
50    pub(super) fn node_address(&self) -> (SurfaceIndex, Option<NodeIndex>) {
51        match *self {
52            TreeComponent::Surface(surface) => (surface, None),
53            TreeComponent::Node(dst_surf, dst_node) => (dst_surf, Some(dst_node)),
54            TreeComponent::Tab(dst_surf, dst_node, _) => (dst_surf, Some(dst_node)),
55        }
56    }
57
58    pub(super) fn surface_address(&self) -> SurfaceIndex {
59        match *self {
60            TreeComponent::Surface(surface)
61            | TreeComponent::Node(surface, _)
62            | TreeComponent::Tab(surface, _, _) => surface,
63        }
64    }
65
66    pub(super) fn is_surface(&self) -> bool {
67        matches!(self, TreeComponent::Surface(_))
68    }
69}

88/// Draws one of the Tab drop destination icons inside `rect`, which one you get is specified by `is_top_bottom`.
89/// The smoothing code is from egui_tiles, see
90/// https://github.com/rerun-io/egui_tiles/blob/f86273ba8ff9f44a9817067abbf977ba5cdcb9fa/src/tree.rs#L724-L752
91/// for details
92fn button_ui(
93    response_rect: Rect,
94    target_rect: Rect,
95    ui: &Ui,
96    lock: &mut bool,
97    mouse_pos: Pos2,
98    style: &Style,
99) -> bool {
100    let is_mouse_over = response_rect.contains(mouse_pos);
101    if is_mouse_over && !*lock {
102        draw_overlay_rect(target_rect, ui, style);
103    }
104    lock.bitor_assign(is_mouse_over);
105    is_mouse_over
106}

135    pub(super) fn resolve_icon_based(
136        &mut self,
137        ui: &Ui,
138        style: &Style,
139        allowed_splits: AllowedSplits,
140        windows_allowed: bool,
141        window_bounds: Rect,
142    ) -> Option<TabDestination> {
143        assert!(!self.is_on_title_bar());
144
145        draw_highlight_rect(self.hover.rect, ui, style);
146        let mut hovering_buttons = false;
147        let (rect, pointer) = (self.hover.rect, self.pointer);
148        let rect = {
149            let mut rect = rect;
150            rect.min.y += style.tab_bar.height;
151            rect = rect.shrink(10.0);
152            rect
153        };
154        let mut destination: Option<TabDestination> = windows_allowed
155            .then(|| TabDestination::Window(Rect::from_min_size(pointer, self.drag.rect.size())));
156
157        let center = rect.center();
158
159        if button_ui(
160            Rect::from_center_size(center, rect.size() / 3.0),
161            rect,
162            ui,
163            &mut hovering_buttons,
164            pointer,
165            style,
166        ) {
167            match self.hover.dst {
168                TreeComponent::Node(surface, node) => {
169                    destination = Some(TabDestination::Node(surface, node, TabInsert::Append))
170                }
171                TreeComponent::Surface(surface) => {
172                    destination = Some(TabDestination::EmptySurface(surface))
173                }
174                _ => (),
175            }
176        }
177
178        for split in [Split::Below, Split::Right, Split::Above, Split::Left] {
179            match allowed_splits {
180                AllowedSplits::TopBottomOnly if !split.is_top_bottom() => continue,
181                AllowedSplits::LeftRightOnly if !split.is_left_right() => continue,
182                AllowedSplits::None => continue,
183                _ => {
184                    // let offset_value = shortest_side + style.overlay.button_spacing;
185                    // Split the screen into nine equal sections:
186                    //
187                    // +---+---+---+
188                    // | L | A | R |
189                    // +---+---+---+
190                    // | L | C | R |
191                    // +---+---+---+
192                    // | L | B | R |
193                    // +---+---+---+
194                    //
195                    let offset_vector = match split {
196                        Split::Above => vec2(0.0, -rect.height() / 3.0),
197                        Split::Below => vec2(0.0, rect.height() / 3.0),
198                        Split::Left => vec2(-rect.width() / 3.0, 0.0),
199                        Split::Right => vec2(rect.width() / 3.0, 0.0),
200                    };
201                    let size_vector = match split {
202                        Split::Above | Split::Below => {
203                            vec2(rect.width() / 3.0, rect.height() / 3.0)
204                        }
205                        Split::Left | Split::Right => vec2(rect.width() / 3.0, rect.height()),
206                    };
207                    let target_rect = match split {
208                        Split::Above => {
209                            let mut rect = rect;
210                            rect.max.y = rect.center().y;
211                            rect
212                        }
213                        Split::Below => {
214                            let mut rect = rect;
215                            rect.min.y = rect.center().y;
216                            rect
217                        }
218                        Split::Left => {
219                            let mut rect = rect;
220                            rect.max.x = rect.center().x;
221                            rect
222                        }
223                        Split::Right => {
224                            let mut rect = rect;
225                            rect.min.x = rect.center().x;
226                            rect
227                        }
228                    };
229                    if button_ui(
230                        Rect::from_center_size(center + offset_vector, size_vector),
231                        target_rect,
232                        ui,
233                        &mut hovering_buttons,
234                        pointer,
235                        style,
236                    ) {
237                        if let TreeComponent::Node(surface, node) = self.hover.dst {
238                            destination =
239                                Some(TabDestination::Node(surface, node, TabInsert::Split(split)))
240                        }
241                    }
242                }
243            }
244        }
245        let hovering_rect = self.hover.rect.contains(pointer);
246        let target_lock_state = match (hovering_rect, hovering_buttons) {
247            (false, false) => LockState::Unlocked,
248            (_, true) => LockState::HardLock,
249            (true, _) => LockState::SoftLock,
250        };
251        self.update_lock(target_lock_state, style, ui.ctx());
252        if let Some(TabDestination::Window(rect)) = destination {
253            let rect = self.window_preview_rect(rect);
254            let rect_bounded = constrain_rect_to_area(ui, rect, window_bounds);
255            draw_overlay_rect(rect_bounded, ui, style);
256        }
257        destination
258    }

260    pub(super) fn resolve_traditional(
261        &mut self,
262        ui: &Ui,
263        style: &Style,
264        allowed_splits: AllowedSplits,
265        windows_allowed: bool,
266        window_bounds: Rect,
267    ) -> Option<TabDestination> {
268        // If windows are not allowed, any hover over a window is immediately disallowed.
269        if !windows_allowed && self.hover.dst.surface_address() != SurfaceIndex::main() {
270            return None;
271        }
272        draw_highlight_rect(self.hover.rect, ui, style);
273
274        // Deals with hovers over tab bar and tab titles.
275        if let Some(rect) = self.hover.tab {
276            draw_overlay_rect(rect, ui, style);
277            let target_lock_state = if rect.contains(self.pointer) {
278                LockState::SoftLock
279            } else {
280                LockState::Unlocked
281            };
282            self.update_lock(target_lock_state, style, ui.ctx());
283            return Some(self.hover.dst.as_tab_destination());
284        }
285
286        // Main cases, splits, window creations, etc.
287        let (hover_rect, pointer) = (self.hover.rect, self.pointer);
288        let center = hover_rect.center();
289
290        let (tab_insertion, overlay_rect) = {
291            // A reverse lerp of the pointers position relative to the hovered leaf rect.
292            // Range is (-0.5, -0.5) to (0.5, 0.5)
293            let a_pos = (Pos2::new(
294                inverse_lerp(hover_rect.x_range().into(), pointer.x).unwrap(),
295                inverse_lerp(hover_rect.y_range().into(), pointer.y).unwrap(),
296            ) - Pos2::new(0.5, 0.5))
297            .to_pos2();
298
299            let center_drop_rect = Rect::from_center_size(
300                Pos2::ZERO,
301                Vec2::splat(style.overlay.feel.center_drop_coverage),
302            );
303            let window_drop_rect = Rect::from_center_size(
304                Pos2::ZERO,
305                Vec2::splat(style.overlay.feel.window_drop_coverage),
306            );
307
308            // Find out what kind of tab insertion (if any) should be used to move this widget.
309            if center_drop_rect.contains(a_pos) {
310                (Some(TabInsert::Append), Rect::EVERYTHING)
311            } else if window_drop_rect.contains(a_pos) {
312                match windows_allowed {
313                    true => (None, Rect::NOTHING),
314                    false => (Some(TabInsert::Append), Rect::EVERYTHING),
315                }
316            } else {
317                // Assessing if were above/below the two linear functions x-y=0 and -x-y=0 determines
318                // what "diagonal" quadrant were in.
319                let a_pos = match allowed_splits {
320                    AllowedSplits::All => a_pos,
321                    AllowedSplits::LeftRightOnly => Pos2::new(a_pos.x, 0.0),
322                    AllowedSplits::TopBottomOnly => Pos2::new(0.0, a_pos.y),
323                    AllowedSplits::None => Pos2::ZERO,
324                };
325                if a_pos == Pos2::ZERO {
326                    match windows_allowed {
327                        true => (None, Rect::NOTHING),
328                        false => (Some(TabInsert::Append), Rect::EVERYTHING),
329                    }
330                } else {
331                    match (a_pos.x - a_pos.y > 0., -a_pos.x - a_pos.y > 0.) {
332                        (true, true) => (
333                            Some(TabInsert::Split(Split::Above)),
334                            Rect::everything_above(center.y),
335                        ),
336                        (false, true) => (
337                            Some(TabInsert::Split(Split::Left)),
338                            Rect::everything_left_of(center.x),
339                        ),
340                        (true, false) => (
341                            Some(TabInsert::Split(Split::Right)),
342                            Rect::everything_right_of(center.x),
343                        ),
344                        (false, false) => (
345                            Some(TabInsert::Split(Split::Below)),
346                            Rect::everything_below(center.y),
347                        ),
348                    }
349                }
350            }
351        };
352
353        let default_value = windows_allowed
354            .then(|| TabDestination::Window(Rect::from_min_size(pointer, self.drag.rect.size())));
355        let final_result = tab_insertion.map_or(default_value, |tab| match self.hover.dst {
356            TreeComponent::Surface(surface) => Some(TabDestination::EmptySurface(surface)),
357            TreeComponent::Node(surface, node) => Some(TabDestination::Node(surface, node, tab)),
358            _ => None,
359        });
360
361        self.update_lock(LockState::SoftLock, style, ui.ctx());
362
363        // Draw the overlay
364        match final_result {
365            Some(TabDestination::Window(rect)) => {
366                let rect = self.window_preview_rect(rect);
367                let rect_bounded = constrain_rect_to_area(ui, rect, window_bounds);
368                draw_overlay_rect(rect_bounded, ui, style);
369            }
370            Some(_) => {
371                draw_overlay_rect(hover_rect.intersect(overlay_rect), ui, style);
372            }
373            None => (),
374        }
375
376        final_result
377    }

379    fn update_lock(&mut self, target_state: LockState, style: &Style, ctx: &Context) {
380        match self.locked.as_mut() {
381            Some(lock_time) => {
382                if target_state == LockState::HardLock {
383                    *lock_time = ctx.input(|i| i.time);
384                }
385                let window_hold = if !self.hover.dst.surface_address().is_main() {
386                    ctx.request_repaint();
387                    self.is_locked(style, ctx)
388                } else {
389                    false
390                };
391                if target_state == LockState::Unlocked && !window_hold {
392                    self.locked = None;
393                }
394            }
395            None => {
396                if target_state != LockState::Unlocked {
397                    self.locked = Some(ctx.input(|i| i.time));
398                }
399            }
400        }
401    }

403    pub(super) fn is_locked(&self, style: &Style, ctx: &Context) -> bool {
404        match self.locked.as_ref() {
405            Some(lock_time) => {
406                let elapsed = ctx.input(|i| (i.time - lock_time) as f32);
407                ctx.request_repaint();
408                elapsed < style.overlay.feel.max_preference_time
409            }
410            None => false,
411        }
412    }

423#[inline]
424fn draw_overlay_rect(target_rect: Rect, ui: &Ui, style: &Style) {
425    let smoothed_rect = create_smoothed_rect(target_rect, ui.ctx());
426    let painter = make_overlay_painter(ui);
427    painter.rect(
428        smoothed_rect,
429        style.overlay.selection_corner_radius,
430        style.overlay.selection_color,
431        Stroke::new(
432            style.overlay.selection_stroke_width,
433            style.overlay.selection_color.to_opaque(),
434        ),
435        StrokeKind::Inside,
436    );
437}

439#[inline]
440fn create_smoothed_rect(target_rect: Rect, ctx: &Context) -> Rect {
441    let dt = ctx.input(|input| input.stable_dt).at_most(0.1);
442    let mut requires_repaint = false;
443    let smoothed_rect = ctx.data_mut(|map| {
444        let smoothed: &mut Rect = map.get_temp_mut_or(Id::new("tabs smoothing rect"), target_rect);
445        let t = egui::emath::exponential_smooth_factor(0.9, 0.05, dt);
446        *smoothed = smoothed.lerp_towards(&target_rect, t);
447
448        let diff = smoothed.min.distance(target_rect.min) + smoothed.max.distance(target_rect.max);
449        if diff < 0.5 {
450            *smoothed = target_rect;
451        } else {
452            requires_repaint = true;
453        }
454        *smoothed
455    });
456    if requires_repaint {
457        ctx.request_repaint();
458    };
459    smoothed_rect
460}

462/// An adapted version of the [`egui::Area`]s code for restricting an area rect to a bound.
463fn constrain_rect_to_area(ui: &Ui, rect: Rect, mut bounds: Rect) -> Rect {
464    if rect.width() > bounds.width() {
465        // Allow overlapping side bars.
466        let screen_rect = ui.ctx().content_rect();
467        (bounds.min.x, bounds.max.x) = (screen_rect.min.x, screen_rect.max.x);
468    }
469    if rect.height() > bounds.height() {
470        // Allow overlapping top/bottom bars:
471        let screen_rect = ui.ctx().content_rect();
472        (bounds.min.y, bounds.max.y) = (screen_rect.min.y, screen_rect.max.y);
473    }
474
475    let mut pos = rect.min;
476
477    // Constrain to screen, unless window is too large to fit:
478    let margin_x = (rect.width() - bounds.width()).at_least(0.0);
479    let margin_y = (rect.height() - bounds.height()).at_least(0.0);
480
481    pos.x = pos.x.at_most(bounds.right() + margin_x - rect.width()); // move left if needed
482    pos.x = pos.x.at_least(bounds.left() - margin_x); // move right if needed
483    pos.y = pos.y.at_most(bounds.bottom() + margin_y - rect.height()); // move right if needed
484    pos.y = pos.y.at_least(bounds.top() - margin_y); // move down if needed
485
486    pos = pos.round_to_pixels(ui.painter().pixels_per_point());
487
488    Rect::from_min_size(pos, rect.size())
489}
