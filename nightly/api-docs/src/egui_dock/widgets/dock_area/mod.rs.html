18/// Displays a [`DockState`] in `egui`.
19pub struct DockArea<'tree, Tab> {
20    id: Id,
21    dock_state: &'tree mut DockState<Tab>,
22    style: Option<Style>,
23    show_add_popup: bool,
24    show_add_buttons: bool,
25    show_close_buttons: bool,
26    tab_context_menus: bool,
27    draggable_tabs: bool,
28    show_tab_name_on_hover: bool,
29    show_window_close_buttons: bool,
30    show_window_collapse_buttons: bool,
31    show_leaf_close_all_buttons: bool,
32    show_leaf_collapse_buttons: bool,
33    show_secondary_button_hint: bool,
34    secondary_button_modifiers: Modifiers,
35    secondary_button_on_modifier: bool,
36    secondary_button_context_menu: bool,
37    allowed_splits: AllowedSplits,
38    window_bounds: Option<Rect>,
39
40    to_remove: Vec<TabRemoval>,
41    to_detach: Vec<(SurfaceIndex, NodeIndex, TabIndex)>,
42    new_focused: Option<(SurfaceIndex, NodeIndex)>,
43    tab_hover_rect: Option<(Rect, TabIndex)>,
44}

47impl<'tree, Tab> DockArea<'tree, Tab> {
48    /// Creates a new [`DockArea`] from the provided [`DockState`].
49    #[inline(always)]
50    pub fn new(tree: &'tree mut DockState<Tab>) -> DockArea<'tree, Tab> {
51        Self {
52            id: Id::new("egui_dock::DockArea"),
53            dock_state: tree,
54            style: None,
55            show_add_popup: false,
56            show_add_buttons: false,
57            show_close_buttons: true,
58            tab_context_menus: true,
59            draggable_tabs: true,
60            show_tab_name_on_hover: false,
61            allowed_splits: AllowedSplits::default(),
62            to_remove: Vec::new(),
63            to_detach: Vec::new(),
64            new_focused: None,
65            tab_hover_rect: None,
66            window_bounds: None,
67            show_window_close_buttons: true,
68            show_window_collapse_buttons: true,
69            show_leaf_close_all_buttons: true,
70            show_leaf_collapse_buttons: true,
71            show_secondary_button_hint: true,
72            secondary_button_modifiers: Modifiers::SHIFT,
73            secondary_button_on_modifier: true,
74            secondary_button_context_menu: true,
75        }
76    }
77
78    /// Sets the [`DockArea`] ID. Useful if you have more than one [`DockArea`].
79    #[inline(always)]
80    pub fn id(mut self, id: Id) -> Self {
81        self.id = id;
82        self
83    }
84
85    /// Sets the look and feel of the [`DockArea`].
86    #[inline(always)]

88        self.style = Some(style);
89        self
90    }
91
92    /// Shows or hides the add button popup.

95        self.show_add_popup = show_add_popup;
96        self
97    }
98
99    /// Shows or hides the tab add buttons.

102        self.show_add_buttons = show_add_buttons;
103        self
104    }
105
106    /// Shows or hides the tab close buttons.

109        self.show_close_buttons = show_close_buttons;

111    }

113    /// Whether tabs show a context menu when right-clicked.

118    }
