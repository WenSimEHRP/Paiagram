23    pub(super) fn show_leaf(
24        &mut self,
25        ui: &mut Ui,
26        state: &mut State,
27        (surface_index, node_index): (SurfaceIndex, NodeIndex),
28        tab_viewer: &mut impl TabViewer<Tab = Tab>,
29        fade_style: Option<(&Style, f32)>,
30    ) {
31        assert!(self.dock_state[surface_index][node_index].is_leaf());
32        let collapsed = self.dock_state[surface_index][node_index].is_collapsed();
33
34        let rect = self.dock_state[surface_index][node_index]
35            .rect()
36            .expect("This node must be a leaf");
37        let ui = &mut ui.new_child(
38            UiBuilder::new()
39                .max_rect(rect)
40                .layout(Layout::top_down_justified(Align::Min))
41                .id_salt((node_index, "node")),
42        );
43        let spacing = ui.spacing().item_spacing;
44        ui.spacing_mut().item_spacing = Vec2::ZERO;
45        ui.set_clip_rect(rect);
46
47        if self.dock_state[surface_index][node_index].tabs_count() == 0 {
48            return;
49        }
50        let tabbar_rect = self.tab_bar(
51            ui,
52            state,
53            (surface_index, node_index),
54            tab_viewer,
55            fade_style.map(|(style, _)| style),
56            collapsed,
57        );
58        self.tab_body(
59            ui,
60            state,
61            (surface_index, node_index),
62            tab_viewer,
63            spacing,
64            tabbar_rect,
65            fade_style,
66            collapsed,
67        );
68
69        let tabs = self.dock_state[surface_index][node_index]
70            .tabs_mut()
71            .expect("This node must be a leaf here");
72        for (tab_index, tab) in tabs.iter_mut().enumerate() {
73            if tab_viewer.force_close(tab) {
74                self.to_remove.push(TabRemoval::Tab(
75                    surface_index,
76                    node_index,
77                    TabIndex(tab_index),
78                    ForcedRemoval(true),
79                ));
80            }
81        }
82    }

84    fn tab_bar(
85        &mut self,
86        ui: &mut Ui,
87        state: &mut State,
88        (surface_index, node_index): (SurfaceIndex, NodeIndex),
89        tab_viewer: &mut impl TabViewer<Tab = Tab>,
90        fade_style: Option<&Style>,
91        collapsed: bool,
92    ) -> Rect {
93        assert!(self.dock_state[surface_index][node_index].is_leaf());
94
95        let style = fade_style.unwrap_or_else(|| self.style.as_ref().unwrap());
96        let (tabbar_outer_rect, tabbar_response) = ui.allocate_exact_size(
97            vec2(ui.available_width(), style.tab_bar.height),
98            Sense::hover(),
99        );
100        ui.painter().rect_filled(
101            tabbar_outer_rect,
102            style.tab_bar.corner_radius,
103            style.tab_bar.bg_fill,
104        );
105
106        let tabbar_outer_rect = tabbar_outer_rect - style.tab_bar.inner_margin;
107
108        let mut available_width = tabbar_outer_rect.width();
109        let scroll_bar_width = available_width;
110        if available_width == 0.0 {
111            return tabbar_outer_rect;
112        }
113
114        // Reserve space for the buttons at the ends of the tab bar.
115
116        if self.show_add_buttons {
117            available_width -= Style::TAB_ADD_BUTTON_SIZE;
118        }
119
120        if self.show_leaf_close_all_buttons {
121            available_width -= Style::TAB_CLOSE_ALL_BUTTON_SIZE;
122        }
123
124        if self.show_leaf_collapse_buttons {
125            available_width -= Style::TAB_COLLAPSE_BUTTON_SIZE;
126        }
127
128        let (actual_width, tab_hovered) = {
129            let leaf = self.dock_state[surface_index][node_index]
130                .get_leaf_mut()
131                .expect("This node must be a leaf");
132
133            let tabbar_inner_rect = Rect::from_min_size(
134                (tabbar_outer_rect.min - pos2(-leaf.scroll, 0.0)
135                    + vec2(
136                        if self.show_leaf_collapse_buttons {
137                            Style::TAB_COLLAPSE_BUTTON_SIZE
138                        } else {
139                            0.0
140                        },
141                        0.0,
142                    ))
143                .to_pos2(),
144                vec2(tabbar_outer_rect.width(), tabbar_outer_rect.height()),
145            );
146
147            let tabs_ui = &mut ui.new_child(
148                UiBuilder::new()
149                    .max_rect(tabbar_inner_rect)
150                    .layout(Layout::left_to_right(Align::Center))
151                    .id_salt("tabs"),
152            );
153
154            let mut clip_rect = tabbar_outer_rect;
155            clip_rect.set_width(available_width);
156            if self.show_leaf_collapse_buttons {
157                clip_rect = clip_rect.translate(vec2(Style::TAB_COLLAPSE_BUTTON_SIZE, 0.0));
158            }
159            tabs_ui.set_clip_rect(clip_rect);
160
161            // Desired size for tabs in "expanded" mode.
162            let prefered_width = style
163                .tab_bar
164                .fill_tab_bar
165                .then_some(available_width / (leaf.tabs.len() as f32));
166
167            let tab_hovered = self.tabs(
168                tabs_ui,
169                state,
170                (surface_index, node_index),
171                tab_viewer,
172                tabbar_outer_rect,
173                prefered_width,
174                fade_style,
175            );
176
177            // Draw hline from tab end to edge of tab bar.
178            let px = ui.ctx().pixels_per_point().recip();
179            let style = fade_style.unwrap_or_else(|| self.style.as_ref().unwrap());
180
181            ui.painter().hline(
182                tabs_ui.min_rect().right().min(clip_rect.right())..=tabbar_outer_rect.right(),
183                tabbar_outer_rect.bottom() - px,
184                (px, style.tab_bar.hline_color),
185            );
186
187            // Add button at the ends of the tab bar.
188            if self.show_add_buttons {
189                let offset = match style.buttons.add_tab_align {
190                    TabAddAlign::Left => {
191                        (clip_rect.width() - tabs_ui.min_rect().width()).at_least(0.0)
192                    }
193                    TabAddAlign::Right => 0.0,
194                } + if self.show_leaf_close_all_buttons {
195                    Style::TAB_CLOSE_ALL_BUTTON_SIZE
196                } else {
197                    0.0
198                };
199                self.tab_plus(
200                    ui,
201                    surface_index,
202                    node_index,
203                    tab_viewer,
204                    tabbar_outer_rect,
205                    offset,
206                    fade_style,
207                );
208            }
209
210            if self.show_leaf_close_all_buttons {
211                // Current leaf contains non-closable tabs.
212                let disabled = self.dock_state[surface_index][node_index]
213                    .get_leaf_mut()
214                    .map(|leaf| !leaf.tabs.iter_mut().all(|tab| tab_viewer.is_closeable(tab)))
215                    .expect("This node must be a leaf");
216
217                // Current window contains non-closable tabs.
218                let close_window_disabled = disabled
219                    || !self.dock_state[surface_index].iter_mut().all(|node| {
220                        node.get_leaf_mut().is_none_or(|leaf| {
221                            leaf.tabs.iter_mut().all(|tab| tab_viewer.is_closeable(tab))
222                        })
223                    });
224
225                self.tab_close_all(
226                    ui,
227                    surface_index,
228                    node_index,
229                    tabbar_outer_rect,
230                    fade_style,
231                    disabled,
232                    close_window_disabled,
233                )
234            }
235
236            if self.show_leaf_collapse_buttons {
237                self.tab_collapse(
238                    ui,
239                    surface_index,
240                    node_index,
241                    tabbar_outer_rect,
242                    fade_style,
243                    collapsed,
244                )
245            }
246
247            (tabs_ui.min_rect().width(), tab_hovered)
248        };
249
250        self.tab_bar_scroll(
251            ui,
252            state,
253            (surface_index, node_index),
254            actual_width,
255            available_width,
256            scroll_bar_width,
257            &tabbar_response,
258            tab_hovered,
259            fade_style,
260        );
261
262        tabbar_outer_rect
263    }

265    #[allow(clippy::too_many_arguments)]
266    fn tabs(
267        &mut self,
268        tabs_ui: &mut Ui,
269        state: &mut State,
270        (surface_index, node_index): (SurfaceIndex, NodeIndex),
271        tab_viewer: &mut impl TabViewer<Tab = Tab>,
272        tabbar_outer_rect: Rect,
273        preferred_width: Option<f32>,
274        fade: Option<&Style>,
275    ) -> bool {
276        let mut tab_hovered = false;
277
278        assert!(self.dock_state[surface_index][node_index].is_leaf());
279
280        let focused = self.dock_state.focused_leaf();
281        let tabs_len = {
282            let tabs = self.dock_state[surface_index][node_index]
283                .tabs()
284                .expect("This node must be a leaf here");
285            tabs.len()
286        };
287
288        for tab_index in 0..tabs_len {
289            let id = self
290                .id
291                .with((surface_index, "surface"))
292                .with((node_index, "node"))
293                .with((tab_index, "tab"));
294            let tab_index = TabIndex(tab_index);
295            let is_being_dragged = tabs_ui.ctx().is_being_dragged(id)
296                && tabs_ui.input(|i| i.pointer.is_decidedly_dragging())
297                && self.draggable_tabs;
298
299            if is_being_dragged {
300                tabs_ui.output_mut(|o| o.cursor_icon = CursorIcon::Grabbing);
301            }
302
303            let (is_active, label, tab_style, closeable) = {
304                let leaf = self.dock_state[surface_index][node_index]
305                    .get_leaf_mut()
306                    .expect("This node must be a leaf");
307                let style = fade.unwrap_or_else(|| self.style.as_ref().unwrap());
308                let tab_style = tab_viewer.tab_style_override(&leaf.tabs[tab_index.0], &style.tab);
309                (
310                    leaf.active == tab_index || is_being_dragged,
311                    tab_viewer.title(&mut leaf.tabs[tab_index.0]),
312                    tab_style.unwrap_or(style.tab.clone()),
313                    tab_viewer.is_closeable(&leaf.tabs[tab_index.0]),
314                )
315            };
316
317            let show_close_button = self.show_close_buttons && closeable;
318
319            let (response, title_id) = if is_being_dragged {
320                let layer_id = LayerId::new(Order::Tooltip, id);
321                let response = tabs_ui
322                    .scope_builder(UiBuilder::new().layer_id(layer_id), |ui| {
323                        self.tab_title(
324                            ui,
325                            &tab_style,
326                            id,
327                            label,
328                            is_active && Some((surface_index, node_index)) == focused,
329                            is_active,
330                            is_being_dragged,
331                            preferred_width,
332                            show_close_button,
333                            fade,
334                        )
335                    })
336                    .response;
337                let title_id = response.id;
338
339                let response =
340                    tabs_ui.interact(response.rect, id.with("dragged"), Sense::click_and_drag());
341
342                if let Some(pointer_pos) = tabs_ui.ctx().pointer_interact_pos() {
343                    let start = *state.drag_start.get_or_insert(pointer_pos);
344                    let delta = pointer_pos - start;
345                    if delta.x.abs() > 30.0 || delta.y.abs() > 6.0 {
346                        tabs_ui
347                            .ctx()
348                            .transform_layer_shapes(layer_id, TSTransform::new(delta, 1.0));
349
350                        tabs_ui.memory_mut(|mem| {
351                            mem.data.insert_temp(
352                                self.id.with("drag_data"),
353                                Some(DragData {
354                                    src: TreeComponent::Tab(surface_index, node_index, tab_index),
355                                    rect: self.dock_state[surface_index][node_index]
356                                        .rect()
357                                        .unwrap(),
358                                }),
359                            );
360                        });
361                    }
362                }
363
364                (response, title_id)
365            } else {
366                if tab_index.0 != 0 {
367                    tabs_ui.allocate_space(vec2(tab_style.spacing, 0.0));
368                }
369                let (mut response, close_response) = self.tab_title(
370                    tabs_ui,
371                    &tab_style,
372                    id,
373                    label,
374                    is_active && Some((surface_index, node_index)) == focused,
375                    is_active,
376                    is_being_dragged,
377                    preferred_width,
378                    show_close_button,
379                    fade,
380                );
381                let title_id = response.id;
382                let close_clicked = close_response.is_some_and(|res| res.clicked());
383                let is_lonely_tab = self.dock_state[surface_index].num_tabs() == 1;
384
385                if self.show_tab_name_on_hover {
386                    let tabs = self.dock_state[surface_index][node_index]
387                        .tabs_mut()
388                        .expect("This node must be a leaf");
389                    let tab = &mut tabs[tab_index.0];
390                    response = response.on_hover_ui(|ui| {
391                        ui.label(tab_viewer.title(tab));
392                    });
393                }
394
395                if self.tab_context_menus {
396                    let eject_button =
397                        Button::new(&self.dock_state.translations.tab_context_menu.eject_button);
398                    let close_button =
399                        Button::new(&self.dock_state.translations.tab_context_menu.close_button);
400
401                    response.context_menu(|ui| {
402                        let leaf = self.dock_state[surface_index][node_index]
403                            .get_leaf_mut()
404                            .expect("This node must be a leaf");
405                        let tab = &mut leaf.tabs[tab_index.0];
406
407                        tab_viewer.context_menu(ui, tab, surface_index, node_index);
408                        if (surface_index.is_main() || !is_lonely_tab)
409                            && tab_viewer.allowed_in_windows(tab)
410                            && ui.add(eject_button).clicked()
411                        {
412                            self.to_detach.push((surface_index, node_index, tab_index));
413                            ui.close();
414                        }
415                        if show_close_button && ui.add(close_button).clicked() {
416                            match tab_viewer.on_close(tab) {
417                                OnCloseResponse::Close => self.to_remove.push(TabRemoval::Tab(
418                                    surface_index,
419                                    node_index,
420                                    tab_index,
421                                    ForcedRemoval(false),
422                                )),
423                                OnCloseResponse::Focus => {
424                                    leaf.active = tab_index;
425                                    self.new_focused = Some((surface_index, node_index));
426                                }
427                                OnCloseResponse::Ignore => (),
428                            }
429                            ui.close();
430                        }
431                    });
432                }
433
434                if close_clicked {
435                    self.to_remove.push(TabRemoval::Tab(
436                        surface_index,
437                        node_index,
438                        tab_index,
439                        ForcedRemoval(false),
440                    ));
441                }
442
443                if let Some(pos) = state.last_hover_pos {
444                    // Use response.rect.contains instead of
445                    // response.hovered as the dragged tab covers
446                    // the underlying tab
447                    if state.drag_start.is_some() && response.rect.contains(pos) {
448                        self.tab_hover_rect = Some((response.rect, tab_index));
449                    }
450                }
451
452                (response, title_id)
453            };
454
455            if response.hovered() {
456                tab_hovered = true;
457            }
458
459            // Paint hline below each tab unless its active (or option says otherwise).
460            let leaf = self.dock_state[surface_index][node_index]
461                .get_leaf_mut()
462                .unwrap();
463            let tab = &mut leaf.tabs[tab_index.0];
464            let style = fade.unwrap_or_else(|| self.style.as_ref().unwrap());
465            let tab_style = tab_viewer.tab_style_override(tab, &style.tab);
466            let tab_style = tab_style.as_ref().unwrap_or(&style.tab);
467
468            if !is_active || tab_style.hline_below_active_tab_name {
469                let px = tabs_ui.ctx().pixels_per_point().recip();
470                tabs_ui.painter().hline(
471                    response.rect.x_range(),
472                    tabbar_outer_rect.bottom() - px,
473                    (px, style.tab_bar.hline_color),
474                );
475            }
476
477            if response.clicked()
478                || (tabs_ui.memory(|m| m.has_focus(title_id))
479                    && tabs_ui.input(|i| i.key_pressed(Key::Enter) || i.key_pressed(Key::Space)))
480            {
481                leaf.active = tab_index;
482                self.new_focused = Some((surface_index, node_index));
483            }
484
485            tab_viewer.on_tab_button(tab, &response);
486
487            if self.show_close_buttons && tab_viewer.is_closeable(tab) && response.middle_clicked()
488            {
489                self.to_remove.push(TabRemoval::Tab(
490                    surface_index,
491                    node_index,
492                    tab_index,
493                    ForcedRemoval(false),
494                ));
495            }
496        }
497
498        tab_hovered
499    }

501    /// Draws the tab add button.
502    #[allow(clippy::too_many_arguments)]
503    fn tab_plus(
504        &mut self,
505        ui: &mut Ui,
506        surface_index: SurfaceIndex,
507        node_index: NodeIndex,
508        tab_viewer: &mut impl TabViewer<Tab = Tab>,
509        tabbar_outer_rect: Rect,
510        offset: f32,
511        fade_style: Option<&Style>,
512    ) {
513        let rect = Rect::from_min_max(
514            tabbar_outer_rect.right_top() - vec2(Style::TAB_ADD_BUTTON_SIZE + offset, 0.0),
515            tabbar_outer_rect.right_bottom() - vec2(offset, 2.0),
516        );
517
518        let ui = &mut ui.new_child(
519            UiBuilder::new()
520                .max_rect(rect)
521                .layout(Layout::left_to_right(Align::Center))
522                .id_salt((node_index, "tab_add")),
523        );
524
525        let (rect, mut response) = ui.allocate_exact_size(ui.available_size(), Sense::click());
526
527        response = response.on_hover_cursor(CursorIcon::PointingHand);
528
529        let style = fade_style.unwrap_or_else(|| self.style.as_ref().unwrap());
530        let color = if response.hovered() || response.has_focus() {
531            ui.painter()
532                .rect_filled(rect, CornerRadius::ZERO, style.buttons.add_tab_bg_fill);
533            style.buttons.add_tab_active_color
534        } else {
535            style.buttons.add_tab_color
536        };
537
538        let mut plus_rect = rect;
539
540        rect_set_size_centered(&mut plus_rect, Vec2::splat(Style::TAB_ADD_PLUS_SIZE));
541
542        ui.painter().line_segment(
543            [plus_rect.center_top(), plus_rect.center_bottom()],
544            Stroke::new(1.0, color),
545        );
546        ui.painter().line_segment(
547            [plus_rect.right_center(), plus_rect.left_center()],
548            Stroke::new(1.0, color),
549        );
550
551        // Draw button left border.
552        ui.painter().vline(
553            rect.left(),
554            rect.y_range(),
555            Stroke::new(
556                ui.ctx().pixels_per_point().recip(),
557                style.buttons.add_tab_border_color,
558            ),
559        );
560
561        let popup_id = ui.id().with("tab_add_popup");
562        if self.show_add_popup {
563            Popup::menu(&response)
564                .id(popup_id)
565                .close_behavior(PopupCloseBehavior::CloseOnClickOutside)
566                .show(|ui| {
567                    tab_viewer.add_popup(ui, surface_index, node_index);
568                });
569        }
570
571        if response.clicked() {
572            tab_viewer.on_add(surface_index, node_index);
573        }
574    }

576    /// Draws the close all button.
577    #[allow(clippy::too_many_arguments)]
578    #[allow(unused_assignments)]
579    fn tab_close_all(
580        &mut self,
581        ui: &mut Ui,
582        surface_index: SurfaceIndex,
583        node_index: NodeIndex,
584        tabbar_outer_rect: Rect,
585        fade_style: Option<&Style>,
586        disabled: bool,
587        close_window_disabled: bool,
588    ) {
589        let rect = Rect::from_min_max(
590            tabbar_outer_rect.right_top() - vec2(Style::TAB_CLOSE_ALL_BUTTON_SIZE, 0.0),
591            tabbar_outer_rect.right_bottom() - vec2(0.0, 2.0),
592        );
593
594        let ui = &mut ui.new_child(
595            UiBuilder::new()
596                .max_rect(rect)
597                .layout(Layout::left_to_right(Align::Center))
598                .id_salt((node_index, "tab_close_all")),
599        );
600
601        let (rect, mut response) = ui.allocate_exact_size(ui.available_size(), Sense::click());
602
603        let style = fade_style.unwrap_or_else(|| self.style.as_ref().unwrap());
604
605        // Whether we're on "secondary button mode" due to modifier keys
606        let on_secondary_button = self.is_on_secondary_button(surface_index, ui, &response);
607
608        let mut stroke_color = if disabled {
609            style.buttons.close_all_tabs_disabled_color
610        } else if response.hovered() || response.has_focus() {
611            if !(close_window_disabled && on_secondary_button) {
612                ui.painter().rect_filled(
613                    rect,
614                    CornerRadius::ZERO,
615                    style.buttons.close_all_tabs_bg_fill,
616                );
617            }
618            style.buttons.close_all_tabs_active_color
619        } else {
620            style.buttons.close_all_tabs_color
621        };
622
623        let mut close_all_rect = rect;
624
625        rect_set_size_centered(&mut close_all_rect, Vec2::splat(Style::TAB_CLOSE_ALL_SIZE));
626
627        if !disabled {
628            response = response.on_hover_cursor(CursorIcon::PointingHand);
629        }
630
631        if on_secondary_button {
632            // Close the entire window
633            if close_window_disabled {
634                stroke_color = style.buttons.close_all_tabs_disabled_color;
635                response = response
636                    .on_hover_cursor(CursorIcon::NotAllowed)
637                    .on_hover_text(
638                        self.dock_state
639                            .translations
640                            .leaf
641                            .close_all_button_disabled_tooltip
642                            .as_str(),
643                    );
644            }
645            Self::draw_close_window_symbol(ui, stroke_color, close_all_rect);
646        } else {
647            // Close all tabs in this leaf
648            if !disabled {
649                if !surface_index.is_main() && self.secondary_button_context_menu {
650                    response.context_menu(|ui| {
651                        ui.add_enabled_ui(!close_window_disabled, |ui| {
652                            if ui
653                                .button(&self.dock_state.translations.leaf.close_all_button)
654                                .on_disabled_hover_text(
655                                    self.dock_state
656                                        .translations
657                                        .leaf
658                                        .close_all_button_disabled_tooltip
659                                        .as_str(),
660                                )
661                                .clicked()
662                            {
663                                self.to_remove.push(TabRemoval::Window(surface_index));
664                            }
665                        });
666                    });
667                }
668            } else {
669                response = response
670                    .on_hover_cursor(CursorIcon::NotAllowed)
671                    .on_hover_text(
672                        self.dock_state
673                            .translations
674                            .leaf
675                            .close_button_disabled_tooltip
676                            .as_str(),
677                    );
678            }
679
680            if response.clicked() {
681                if on_secondary_button {
682                    if !close_window_disabled {
683                        self.to_remove.push(TabRemoval::Window(surface_index));
684                    }
685                } else if !disabled {
686                    self.to_remove
687                        .push(TabRemoval::Node(surface_index, node_index));
688                }
689            }
690
691            ui.painter().line_segment(
692                [close_all_rect.left_top(), close_all_rect.right_bottom()],
693                Stroke::new(1.0, stroke_color),
694            );
695            ui.painter().line_segment(
696                [close_all_rect.right_top(), close_all_rect.left_bottom()],
697                Stroke::new(1.0, stroke_color),
698            );
699        }
700
701        // Draw button left border.
702        ui.painter().vline(
703            rect.left(),
704            rect.y_range(),
705            Stroke::new(
706                ui.ctx().pixels_per_point().recip(),
707                style.buttons.close_all_tabs_border_color,
708            ),
709        );
710
711        if !disabled && !on_secondary_button {
712            response = self.show_tooltip_hints(surface_index, response);
713        }
714    }

716    /// Draws the collapse button.
717    fn tab_collapse(
718        &mut self,
719        ui: &mut Ui,
720        surface_index: SurfaceIndex,
721        node_index: NodeIndex,
722        tabbar_outer_rect: Rect,
723        fade_style: Option<&Style>,
724        collapsed: bool,
725    ) {
726        let rect = Rect::from_min_max(
727            tabbar_outer_rect.left_top(),
728            tabbar_outer_rect.left_bottom() + vec2(Style::TAB_COLLAPSE_BUTTON_SIZE, 0.0),
729        );
730
731        let ui = &mut ui.new_child(
732            UiBuilder::new()
733                .max_rect(rect)
734                .layout(Layout::left_to_right(Align::Center))
735                .id_salt((node_index, "tab_collapse")),
736        );
737
738        let (rect, mut response) = ui.allocate_exact_size(ui.available_size(), Sense::click());
739
740        response = response.on_hover_cursor(CursorIcon::PointingHand);
741
742        let style = fade_style.unwrap_or_else(|| self.style.as_ref().unwrap());
743
744        // Whether we're on "secondary button mode" due to modifier keys
745        let on_secondary_button = self.is_on_secondary_button(surface_index, ui, &response);
746
747        let color = if response.hovered() || response.has_focus() {
748            ui.painter().rect_filled(
749                rect,
750                CornerRadius::ZERO,
751                style.buttons.collapse_tabs_bg_fill,
752            );
753            style.buttons.collapse_tabs_active_color
754        } else {
755            style.buttons.collapse_tabs_color
756        };
757
758        let mut arrow_rect = rect;
759        rect_set_size_centered(&mut arrow_rect, Vec2::splat(Style::TAB_COLLAPSE_ARROW_SIZE));
760
761        if on_secondary_button {
762            // Collapse the entire window
763            Self::draw_chevron_down(ui, style, color, arrow_rect);
764        } else {
765            // Draw arrow.
766            Self::draw_arrow(collapsed, ui, color, arrow_rect);
767        }
768
769        // Draw button right border.
770        ui.painter().vline(
771            rect.right(),
772            rect.y_range(),
773            Stroke::new(
774                ui.ctx().pixels_per_point().recip(),
775                style.buttons.collapse_tabs_border_color,
776            ),
777        );
778
779        if response.clicked() {
780            if on_secondary_button {
781                self.window_toggle_minimized(surface_index);
782            } else {
783                self.dock_state[surface_index][node_index].set_collapsed(!collapsed);
784                self.dock_state[surface_index].node_update_collapsed(node_index);
785                self.window_update_collapsed(surface_index, node_index);
786            }
787        }
788
789        if !surface_index.is_main() && self.secondary_button_context_menu {
790            response.context_menu(|ui| {
791                if ui
792                    .button(&self.dock_state.translations.leaf.minimize_button)
793                    .clicked()
794                {
795                    ui.close();
796                    self.window_toggle_minimized(surface_index);
797                }
798            });
799        }
800
801        if !on_secondary_button {
802            self.show_tooltip_hints(surface_index, response);
803        }
804    }

933        let surface = &mut self.dock_state[surface_index];
934        let collapsed = surface[node_index].is_collapsed();

938            }

945            };

947                window_state.set_expanded_height(surface_height);
948            }

952    /// * `active` means "the tab that is opened in the parent panel".
953    /// * `focused` means "the tab that was last interacted with".
954    ///
955    /// Returns the main button response plus the response of the close button, if any.
956    #[allow(clippy::too_many_arguments)]
957    fn tab_title(
958        &mut self,
959        ui: &mut Ui,
960        tab_style: &TabStyle,
961        id: Id,
962        label: WidgetText,
963        focused: bool,
964        active: bool,
965        is_being_dragged: bool,
966        preferred_width: Option<f32>,
967        show_close_button: bool,
968        fade: Option<&Style>,
969    ) -> (Response, Option<Response>) {
970        let style = fade.unwrap_or_else(|| self.style.as_ref().unwrap());
971        let galley = label.into_galley(ui, None, f32::INFINITY, TextStyle::Button);
972        let x_spacing = 8.0;
973        let text_width = galley.size().x + 2.0 * x_spacing;
974        let close_button_size = if show_close_button {
975            Style::TAB_CLOSE_BUTTON_SIZE.min(style.tab_bar.height)
976        } else {
977            0.0
978        };
979
980        // Compute total width of the tab bar.
981        let minimum_width = tab_style
982            .minimum_width
983            .unwrap_or(0.0)
984            .at_least(text_width + close_button_size);
985        let tab_width = preferred_width.unwrap_or(0.0).at_least(minimum_width);
986
987        let (_, tab_rect) = ui.allocate_space(vec2(tab_width, ui.available_height()));
988        let mut response = ui.interact(tab_rect, id, Sense::click_and_drag());
989        if ui.ctx().dragged_id().is_none() && self.draggable_tabs {
990            response = response.on_hover_cursor(CursorIcon::Grab);
991        }
992
993        let tab_style = if focused || is_being_dragged {
994            if response.has_focus() {
995                &tab_style.focused_with_kb_focus
996            } else {
997                &tab_style.focused
998            }
999        } else if active {
1000            if response.has_focus() {
1001                &tab_style.active_with_kb_focus
1002            } else {
1003                &tab_style.active
1004            }
1005        } else if response.hovered() {
1006            &tab_style.hovered
1007        } else if response.has_focus() {
1008            &tab_style.inactive_with_kb_focus
1009        } else {
1010            &tab_style.inactive
1011        };
1012
1013        // Draw the full tab first and then the stroke on top to avoid the stroke
1014        // mixing with the background color.
1015        ui.painter()
1016            .rect_filled(tab_rect, tab_style.corner_radius, tab_style.bg_fill);
1017        let stroke_rect = rect_stroke_box(tab_rect, 1.0);
1018        ui.painter().rect_stroke(
1019            stroke_rect,
1020            tab_style.corner_radius,
1021            Stroke::new(1.0, tab_style.outline_color),
1022            StrokeKind::Inside,
1023        );
1024        if !is_being_dragged {
1025            // Make the tab name area connect with the tab ui area.
1026            ui.painter().hline(
1027                RangeInclusive::new(
1028                    stroke_rect.min.x + f32::max(tab_style.corner_radius.sw.into(), 1.5),
1029                    stroke_rect.max.x - f32::max(tab_style.corner_radius.se.into(), 1.5),
1030                ),
1031                stroke_rect.bottom(),
1032                Stroke::new(2.0, tab_style.bg_fill),
1033            );
1034        }
1035
1036        let mut text_rect = tab_rect;
1037        text_rect.set_width(text_rect.width() - close_button_size);
1038        let text_pos = {
1039            let pos = Align2::CENTER_CENTER.pos_in_rect(&text_rect.shrink2(vec2(x_spacing, 0.0)));
1040            pos - galley.size() / 2.0
1041        };
1042
1043        ui.painter()
1044            .add(TextShape::new(text_pos, galley, tab_style.text_color));
1045
1046        let close_response = show_close_button.then(|| {
1047            let mut close_button_rect = tab_rect;
1048            close_button_rect.set_left(text_rect.right());
1049            close_button_rect =
1050                Rect::from_center_size(close_button_rect.center(), Vec2::splat(close_button_size));
1051
1052            let close_response = ui
1053                .interact(close_button_rect, id.with("close-button"), Sense::click())
1054                .on_hover_cursor(CursorIcon::PointingHand);
1055
1056            let color = if close_response.hovered() || close_response.has_focus() {
1057                style.buttons.close_tab_active_color
1058            } else {
1059                style.buttons.close_tab_color
1060            };
1061
1062            if close_response.hovered() || close_response.has_focus() {
1063                let mut corner_radius = tab_style.corner_radius;
1064                corner_radius.nw = 0;
1065                corner_radius.sw = 0;
1066
1067                ui.painter().rect_filled(
1068                    close_button_rect,
1069                    corner_radius,
1070                    style.buttons.close_tab_bg_fill,
1071                );
1072            }
1073
1074            let mut x_rect = close_button_rect;
1075            rect_set_size_centered(&mut x_rect, Vec2::splat(Style::TAB_CLOSE_X_SIZE));
1076            ui.painter().line_segment(
1077                [x_rect.left_top(), x_rect.right_bottom()],
1078                Stroke::new(1.0, color),
1079            );
1080            ui.painter().line_segment(
1081                [x_rect.right_top(), x_rect.left_bottom()],
1082                Stroke::new(1.0, color),
1083            );
1084
1085            close_response
1086        });
1087
1088        (response, close_response)
1089    }

1091    #[allow(clippy::too_many_arguments)]
1092    fn tab_bar_scroll(
1093        &mut self,
1094        ui: &mut Ui,
1095        state: &State,
1096        (surface_index, node_index): (SurfaceIndex, NodeIndex),
1097        actual_width: f32,
1098        available_width: f32,
1099        scroll_bar_width: f32,
1100        tabbar_response: &Response,
1101        tab_hovered: bool,
1102        fade_style: Option<&Style>,
1103    ) {
1104        assert_ne!(available_width, 0.0);
1105
1106        let leaf = self.dock_state[surface_index][node_index]
1107            .get_leaf_mut()
1108            .expect("This node must be a leaf");
1109        let overflow = (actual_width - available_width).at_least(0.0);
1110        let style = fade_style.unwrap_or_else(|| self.style.as_ref().unwrap());
1111
1112        // Compare to 1.0 and not 0.0 to avoid drawing a scroll bar due
1113        // to floating point precision issue during tab drawing.
1114        if overflow > 1.0 {
1115            if style.tab_bar.show_scroll_bar_on_overflow {
1116                // Draw scroll bar
1117                let bar_height = 7.5;
1118                let (scroll_bar_rect, _scroll_bar_response) = ui.allocate_exact_size(
1119                    vec2(scroll_bar_width, bar_height),
1120                    Sense::click_and_drag(),
1121                );
1122
1123                // Compute scroll bar handle position and size.
1124                let overflow_ratio = actual_width / available_width;
1125                let scroll_ratio = -leaf.scroll / overflow;
1126
1127                let scroll_bar_handle_size = overflow_ratio.recip() * scroll_bar_rect.width();
1128                let scroll_bar_handle_start = lerp(
1129                    scroll_bar_rect.left()..=scroll_bar_rect.right() - scroll_bar_handle_size,
1130                    scroll_ratio,
1131                );
1132                let scroll_bar_handle_rect = Rect::from_min_size(
1133                    pos2(scroll_bar_handle_start, scroll_bar_rect.min.y),
1134                    vec2(scroll_bar_handle_size, bar_height),
1135                );
1136
1137                let scroll_bar_handle_response = ui.interact(
1138                    scroll_bar_handle_rect,
1139                    self.id.with((node_index, "node")),
1140                    Sense::drag(),
1141                );
1142
1143                // Coefficient to apply to input displacements so that we move the scroll by the correct amount.
1144                let points_to_scroll_coefficient =
1145                    overflow / (scroll_bar_rect.width() - scroll_bar_handle_size);
1146
1147                leaf.scroll -=
1148                    scroll_bar_handle_response.drag_delta().x * points_to_scroll_coefficient;
1149
1150                if let Some(pos) = state.last_hover_pos {
1151                    if scroll_bar_rect.contains(pos) {
1152                        leaf.scroll += ui
1153                            .input(|i| i.smooth_scroll_delta.y + i.smooth_scroll_delta.x)
1154                            * points_to_scroll_coefficient;
1155                    }
1156                }
1157
1158                // Draw the bar.
1159                ui.painter()
1160                    .rect_filled(scroll_bar_rect, 0.0, ui.visuals().extreme_bg_color);
1161
1162                ui.painter().rect_filled(
1163                    scroll_bar_handle_rect,
1164                    bar_height / 2.0,
1165                    ui.visuals()
1166                        .widgets
1167                        .style(&scroll_bar_handle_response)
1168                        .bg_fill,
1169                );
1170            }
1171
1172            // Handle user input.
1173            if tabbar_response.hovered() || tab_hovered {
1174                leaf.scroll += ui.input(|i| i.smooth_scroll_delta.y + i.smooth_scroll_delta.x);
1175            }
1176        }
1177
1178        leaf.scroll = leaf.scroll.clamp(-overflow, 0.0);
1179    }

1181    #[allow(clippy::too_many_arguments)]
1182    fn tab_body(
1183        &mut self,
1184        ui: &mut Ui,
1185        state: &State,
1186        (surface_index, node_index): (SurfaceIndex, NodeIndex),
1187        tab_viewer: &mut impl TabViewer<Tab = Tab>,
1188        spacing: Vec2,
1189        tabbar_rect: Rect,
1190        fade: Option<(&Style, f32)>,
1191        collapsed: bool,
1192    ) {
1193        let (body_rect, _body_response) =
1194            ui.allocate_exact_size(ui.available_size_before_wrap(), Sense::hover());
1195
1196        let leaf = self.dock_state[surface_index][node_index]
1197            .get_leaf_mut()
1198            .expect("This node must be a leaf");
1199        let LeafNode {
1200            rect,
1201            viewport,
1202            tabs,
1203            active,
1204            ..
1205        } = leaf;
1206        if !collapsed {
1207            if let Some(tab) = tabs.get_mut(active.0) {
1208                if *viewport != body_rect {
1209                    *viewport = body_rect;
1210                    tab_viewer.on_rect_changed(tab);
1211                }
1212
1213                if ui.input(|i| i.pointer.any_click()) {
1214                    if let Some(pos) = state.last_hover_pos {
1215                        if body_rect.contains(pos)
1216                            && Some(ui.layer_id()) == ui.ctx().layer_id_at(pos)
1217                        {
1218                            self.new_focused = Some((surface_index, node_index));
1219                        }
1220                    }
1221                }
1222
1223                let (style, fade_factor) =
1224                    fade.unwrap_or_else(|| (self.style.as_ref().unwrap(), 1.0));
1225                let tabs_styles = tab_viewer.tab_style_override(tab, &style.tab);
1226
1227                let tabs_style = tabs_styles.as_ref().unwrap_or(&style.tab);
1228
1229                if tab_viewer.clear_background(tab) {
1230                    ui.painter().rect_filled(
1231                        body_rect,
1232                        tabs_style.tab_body.corner_radius,
1233                        tabs_style.tab_body.bg_fill,
1234                    );
1235                }
1236
1237                // Construct a new ui with the correct tab id.
1238                //
1239                // We are forced to use `Ui::new` because other methods (eg: push_id) always mix
1240                // the provided id with their own which would cause tabs to change id when moved
1241                // from node to node.
1242                let id = self.id.with(tab_viewer.id(tab));
1243                ui.ctx().check_for_id_clash(id, body_rect, "a tab with id");
1244                let ui = &mut Ui::new(
1245                    ui.ctx().clone(),
1246                    id,
1247                    UiBuilder::new().max_rect(body_rect).layer_id(ui.layer_id()),
1248                );
1249                ui.set_clip_rect(Rect::from_min_max(ui.cursor().min, ui.clip_rect().max));
1250
1251                // Use initial spacing for ui.
1252                ui.spacing_mut().item_spacing = spacing;
1253
1254                // Offset the background rectangle up to hide the top border behind the clip rect.
1255                // To avoid anti-aliasing lines when the stroke width is not divisible by two, we
1256                // need to calculate the effective anti-aliased stroke width.
1257                let effective_stroke_width = (tabs_style.tab_body.stroke.width / 2.0).ceil() * 2.0;
1258                let tab_body_rect = Rect::from_min_max(
1259                    ui.clip_rect().min - vec2(0.0, effective_stroke_width),
1260                    ui.clip_rect().max,
1261                );
1262                ui.painter().rect_stroke(
1263                    rect_stroke_box(tab_body_rect, tabs_style.tab_body.stroke.width),
1264                    tabs_style.tab_body.corner_radius,
1265                    tabs_style.tab_body.stroke,
1266                    StrokeKind::Inside,
1267                );
1268
1269                ScrollArea::new(tab_viewer.scroll_bars(tab)).show(ui, |ui| {
1270                    Frame::new()
1271                        .inner_margin(tabs_style.tab_body.inner_margin)
1272                        .show(ui, |ui| {
1273                            if fade_factor != 1.0 {
1274                                fade_visuals(ui.visuals_mut(), fade_factor);
1275                            }
1276                            let available_rect = ui.available_rect_before_wrap();
1277                            ui.expand_to_include_rect(available_rect);
1278                            tab_viewer.ui(ui, tab);
1279                        });
1280                });
1281            }
1282        }
1283
1284        // change hover destination
1285        if let Some(pointer) = state.last_hover_pos {
1286            // Prevent borrow checker issues.
1287            let rect = rect.to_owned();
1288
1289            // if the dragged tab isn't allowed in a window,
1290            // it's unnecessary to change the hover state
1291            let is_dragged_valid = match &state.dnd {
1292                Some(DragDropState {
1293                    drag: DragData { src, .. },
1294                    ..
1295                }) => match *src {
1296                    TreeComponent::Tab(d_surf, d_node, d_tab) => {
1297                        if let Node::Leaf(leaf) = &mut self.dock_state[d_surf][d_node] {
1298                            tab_viewer.allowed_in_windows(&mut leaf.tabs[d_tab.0])
1299                                || surface_index == SurfaceIndex::main()
1300                        } else {
1301                            true
1302                        }
1303                    }
1304                    _ => unreachable!("collections of nodes can't be dragged (yet)"),
1305                },
1306                _ => true,
1307            };
1308
1309            // Use rect.contains instead of response.hovered as the dragged tab covers
1310            // the underlying responses.
1311            if state.drag_start.is_some() && rect.contains(pointer) && is_dragged_valid {
1312                let on_title_bar = tabbar_rect.contains(pointer);
1313                let (dst, tab) = {
1314                    match self.tab_hover_rect {
1315                        Some((rect, tab_index)) => (
1316                            TreeComponent::Tab(surface_index, node_index, tab_index),
1317                            Some(rect),
1318                        ),
1319                        None => (
1320                            TreeComponent::Node(surface_index, node_index),
1321                            on_title_bar.then_some(tabbar_rect),
1322                        ),
1323                    }
1324                };
1325
1326                ui.memory_mut(|mem| {
1327                    mem.data.insert_temp(
1328                        self.id.with("hover_data"),
1329                        Some(HoverData { rect, dst, tab }),
1330                    );
1331                });
1332            }
1333        }
1334    }
