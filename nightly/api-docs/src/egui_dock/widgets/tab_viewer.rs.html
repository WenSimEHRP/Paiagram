4/// Defines how a tab should behave and be rendered inside a [`Tree`](crate::Tree).
5pub trait TabViewer {
6    /// The type of tab in which you can store state to be drawn in your tabs.
7    type Tab;
8
9    /// The title to be displayed in the tab bar.
10    fn title(&mut self, tab: &mut Self::Tab) -> WidgetText;
11
12    /// Actual tab content.
13    fn ui(&mut self, ui: &mut Ui, tab: &mut Self::Tab);
14
15    /// Content inside the context menu shown when the tab is right-clicked.
16    ///
17    /// `_surface` and `_node` specify which [`Surface`](crate::Surface) and [`Node`](crate::Node)
18    /// that this particular context menu belongs to.
19    fn context_menu(
20        &mut self,
21        _ui: &mut Ui,
22        _tab: &mut Self::Tab,
23        _surface: SurfaceIndex,
24        _node: NodeIndex,
25    ) {
26    }
27
28    /// Unique ID for this tab.
29    ///
30    /// If not implemented, uses tab title text as an ID source.
31    fn id(&mut self, tab: &mut Self::Tab) -> Id {
32        Id::new(self.title(tab).text())
33    }
34
35    /// Called after each tab button is shown, so you can add a tooltip, check for clicks, etc.
36    fn on_tab_button(&mut self, _tab: &mut Self::Tab, _response: &egui::Response) {}
37
38    /// This is called when the `_tab` gets closed by the user.
39    ///
40    /// Returns an `OnCloseResponse` which determines what happens to the tab after this function gets called.
41    fn on_close(&mut self, _tab: &mut Self::Tab) -> OnCloseResponse {
42        OnCloseResponse::Close
43    }
44
45    /// Returns `true` if the user of your app should be able to close a given `_tab`.
46    ///
47    /// By default, `true` is always returned.
48    fn is_closeable(&self, _tab: &Self::Tab) -> bool {
49        true
50    }
51
52    /// Returns `true` if the user of your app should be able to close a given `_tab`.
53    ///
54    /// By default, `true` is always returned.
55    #[deprecated = "Use the `TabViewer::is_closeable` function instead."]
56    fn closeable(&mut self, _tab: &mut Self::Tab) -> bool {
57        true
58    }
59
60    /// This is called every frame after [`ui`](Self::ui) is called, if the `_tab` is active.
61    ///
62    /// Returns `true` if the tab should be forced to close, `false` otherwise.
63    ///
64    /// In the event this function returns true the tab will be removed without calling `on_close`.
65    fn force_close(&mut self, _tab: &mut Self::Tab) -> bool {
66        false
67    }
68
69    /// This is called when the add button is pressed.
70    ///
71    /// `_surface` and `_node` specify which [`Surface`](crate::Surface) and on which
72    /// [`Node`](crate::Node) this particular add button was pressed.
73    fn on_add(&mut self, _surface: SurfaceIndex, _node: NodeIndex) {}
74
75    /// Called when the rectangle of the tab content changes.
76    ///
77    /// This can happen when the window is resized, panels are docked or undocked,
78    /// or when the layout of the dock area is changed in any way that affects
79    /// the available space for the tab content.
80    ///
81    /// This is useful for tabs that need to adjust their content based on the
82    /// available space.
83    fn on_rect_changed(&mut self, _tab: &mut Self::Tab) {}
84
85    /// Content of the popup under the add button. Useful for selecting what type of tab to add.
86    ///
87    /// This requires that [`DockArea::show_add_buttons`](crate::DockArea::show_add_buttons) and
88    /// [`DockArea::show_add_popup`](crate::DockArea::show_add_popup) are set to `true`.
89    fn add_popup(&mut self, _ui: &mut Ui, _surface: SurfaceIndex, _node: NodeIndex) {}
90
91    /// Sets custom style for given tab.
92    fn tab_style_override(&self, _tab: &Self::Tab, _global_style: &TabStyle) -> Option<TabStyle> {
93        None
94    }
95
96    /// Specifies a tab's ability to be shown in a window.
97    ///
98    /// Returns `false` if this tab should never be turned into a window.
99    fn allowed_in_windows(&self, _tab: &mut Self::Tab) -> bool {
100        true
101    }
102
103    /// Whether the tab body will be cleared with the color specified in
104    /// [`TabBarStyle::bg_fill`](crate::TabBarStyle::bg_fill).
105    fn clear_background(&self, _tab: &Self::Tab) -> bool {
106        true
107    }
108
109    /// Returns `true` if the horizontal and vertical scroll bars will be shown for `tab`.
110    ///
111    /// By default, both scroll bars are shown.
112    fn scroll_bars(&self, _tab: &Self::Tab) -> [bool; 2] {
113        [true, true]
114    }
115}

117/// Determines what happens to a tab when a user attempts to close it.
118#[derive(Copy, Clone, Debug, PartialEq, Eq)]
119pub enum OnCloseResponse {
120    /// Closes the tab.
121    Close,
122    /// Focuses on the tab.
123    Focus,
124    /// Ignores the close request.
125    Ignore,
126}
