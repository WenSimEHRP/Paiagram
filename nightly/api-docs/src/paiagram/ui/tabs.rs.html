46pub trait Navigatable {
47    type XOffset: Into<f64> + From<f64>;
48    type YOffset: Into<f64> + From<f64>;
49    fn zoom_x(&self) -> f32;
50    fn zoom_y(&self) -> f32;
51    fn set_zoom(&mut self, zoom_x: f32, zoom_y: f32);
52    fn offset_x(&self) -> f64;
53    fn offset_y(&self) -> f64;
54    fn set_offset(&mut self, offset_x: f64, offset_y: f64);
55    fn screen_pos_to_xy(&self, pos: egui::Pos2) -> (Self::XOffset, Self::YOffset) {
56        let rect = self.visible_rect();
57        let x_per_screen_unit = self.x_per_screen_unit().into();
58        let y_per_screen_unit = self.y_per_screen_unit().into();
59        let x = self.offset_x() + (pos.x - rect.left()) as f64 * x_per_screen_unit;
60        let y = self.offset_y() + (pos.y - rect.top()) as f64 * y_per_screen_unit;
61        (x.into(), y.into())
62    }
63    fn xy_to_screen_pos(&self, x: Self::XOffset, y: Self::YOffset) -> egui::Pos2 {
64        let rect = self.visible_rect();
65        let x_per_screen_unit = self.x_per_screen_unit().into();
66        let y_per_screen_unit = self.y_per_screen_unit().into();
67        let x = x.into();
68        let y = y.into();
69        let screen_x = rect.left() + ((x - self.offset_x()) / x_per_screen_unit) as f32;
70        let screen_y = rect.top() + ((y - self.offset_y()) / y_per_screen_unit) as f32;
71        egui::Pos2::new(screen_x, screen_y)
72    }
73    fn visible_rect(&self) -> egui::Rect;
74    fn visible_x(&self) -> std::ops::Range<Self::XOffset> {
75        let width = self.visible_rect().width() as f64;
76        let start = self.offset_x();
77        let end = start + width * self.x_per_screen_unit().into();
78        start.into()..end.into()
79    }
80    fn visible_y(&self) -> std::ops::Range<Self::YOffset> {
81        let height = self.visible_rect().height() as f64;
82        let start = self.offset_y();
83        let end = start + height * self.y_per_screen_unit().into();
84        start.into()..end.into()
85    }
86    fn x_per_screen_unit(&self) -> Self::XOffset {
87        (1.0 / self.zoom_x().max(f32::EPSILON) as f64).into()
88    }
89    fn y_per_screen_unit(&self) -> Self::YOffset {
90        (1.0 / self.zoom_y().max(f32::EPSILON) as f64).into()
91    }
92    fn allow_axis_zoom(&self) -> bool {
93        false
94    }
95    /// Handles the navigation of the navigatable component.
96    /// Returns true if there are any user input
97    fn handle_navigation(&mut self, ui: &mut Ui, response: &Response) -> bool {
98        let mut moved = response.dragged();
99        let started_pos = ui
100            .ctx()
101            .input(|i| i.pointer.press_origin().or(i.pointer.hover_pos()));
102        let zoom_delta = if self.allow_axis_zoom() {
103            ui.input(|input| input.zoom_delta_2d())
104        } else {
105            let zoom_delta = ui.input(|input| input.zoom_delta());
106            egui::vec2(zoom_delta, zoom_delta)
107        };
108        let scroll_delta = ui.input(|input| input.smooth_scroll_delta);
109        let zooming = (zoom_delta.x - 1.0).abs() > 0.001 || (zoom_delta.y - 1.0).abs() > 0.001;
110
111        if zooming
112            && ui.ui_contains_pointer()
113            && let Some(pos) = response.hover_pos()
114        {
115            moved |= zooming;
116            let old_zoom_x = self.zoom_x();
117            let old_zoom_y = self.zoom_y();
118            let mut new_zoom_x = old_zoom_x * zoom_delta.x;
119            let mut new_zoom_y = old_zoom_y * zoom_delta.y;
120            let (clamped_x, clamped_y) = self.clamp_zoom(new_zoom_x, new_zoom_y);
121            new_zoom_x = clamped_x;
122            new_zoom_y = clamped_y;
123
124            let rel_pos = (pos - response.rect.min) / response.rect.size();
125            let world_width_before = response.rect.width() as f64 / old_zoom_x as f64;
126            let world_width_after = response.rect.width() as f64 / new_zoom_x as f64;
127            let world_pos_before_x = self.offset_x() + rel_pos.x as f64 * world_width_before;
128            let new_offset_x = world_pos_before_x - rel_pos.x as f64 * world_width_after;
129
130            let world_height_before = response.rect.height() as f64 / old_zoom_y as f64;
131            let world_height_after = response.rect.height() as f64 / new_zoom_y as f64;
132            let world_pos_before_y = self.offset_y() + rel_pos.y as f64 * world_height_before;
133            let new_offset_y = world_pos_before_y - rel_pos.y as f64 * world_height_after;
134
135            self.set_zoom(new_zoom_x, new_zoom_y);
136            self.set_offset(new_offset_x, new_offset_y);
137        }
138        // if ui.ui_contains_pointer() || ui.input(|r| r.any_touches()) {
139        if let Some(started_pos) = started_pos
140            && response.rect.contains(started_pos)
141        {
142            let ticks_per_screen_unit = 1.0 / self.zoom_x() as f64;
143            let pan_delta = response.drag_delta() + scroll_delta;
144            let new_offset_x = self.offset_x() - ticks_per_screen_unit * pan_delta.x as f64;
145            let new_offset_y = self.offset_y() - pan_delta.y as f64 / self.zoom_y() as f64;
146            moved |= scroll_delta.x != 0.0;
147            moved |= scroll_delta.y != 0.0;
148            self.set_offset(new_offset_x, new_offset_y);
149        }
150
151        self.post_navigation(response);
152        return moved;
153    }
154    fn clamp_zoom(&self, zoom_x: f32, zoom_y: f32) -> (f32, f32) {
155        (zoom_x, zoom_y)
156    }
157    fn post_navigation(&mut self, _response: &Response) {}
158}
