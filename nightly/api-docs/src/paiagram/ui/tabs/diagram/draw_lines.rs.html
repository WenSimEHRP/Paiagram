5pub fn draw_station_lines<'a>(
6    vertical_offset: f32,
7    painter: &mut Painter,
8    screen_rect: Rect,
9    zoom: f32,
10    to_draw: impl Iterator<Item = (Entity, f32)>,
11    pixels_per_point: f32,
12    visuals: &Visuals,
13    world: &World, // FIXME: make this a system afterwards
14) {
15    // TODO: implement per-station stroke
16    let stroke = Stroke {
17        width: 0.6,
18        color: visuals.window_stroke().color,
19    };
20    // Guard against invalid zoom
21    for (station_entity, height) in to_draw {
22        let mut draw_height = (height - vertical_offset) * zoom + screen_rect.top();
23        stroke.round_center_to_pixel(pixels_per_point, &mut draw_height);
24        painter.hline(
25            screen_rect.left()..=screen_rect.right(),
26            draw_height,
27            stroke,
28        );
29        let layout = painter.layout_no_wrap(
30            world
31                .get::<Name>(station_entity)
32                .map_or("<Unknown>".to_string(), Name::to_string),
33            egui::FontId::proportional(13.0),
34            visuals.text_color(),
35        );
36        let layout_pos = Pos2 {
37            x: screen_rect.left(),
38            y: draw_height - layout.size().y,
39        };
40        painter.galley(layout_pos, layout, visuals.text_color());
41    }
42}

44pub(super) fn ticks_to_screen_x(
45    ticks: i64,
46    screen_rect: Rect,
47    ticks_per_screen_unit: f64,
48    offset_ticks: i64,
49) -> f32 {
50    let base = (ticks - offset_ticks) as f64 / ticks_per_screen_unit;
51    screen_rect.left() + base as f32
52}

54/// Draw vertical time lines and labels
55pub fn draw_time_lines(
56    tick_offset: i64,
57    painter: &mut Painter,
58    screen_rect: Rect,
59    ticks_per_screen_unit: f64,
60    visible_ticks: &std::ops::Range<i64>,
61    pixels_per_point: f32,
62) {
63    const MAX_SCREEN_WIDTH: f64 = 64.0;
64    const MIN_SCREEN_WIDTH: f64 = 32.0;
65    let sizes = [
66        Tick::from_timetable_time(TimetableTime(1)).0, // 1 second
67        Tick::from_timetable_time(TimetableTime(10)).0, // 10 seconds
68        Tick::from_timetable_time(TimetableTime(30)).0, // 30 seconds
69        Tick::from_timetable_time(TimetableTime(60)).0, // 1 minute
70        Tick::from_timetable_time(TimetableTime(60 * 5)).0, // 5 minutes
71        Tick::from_timetable_time(TimetableTime(60 * 10)).0, // 10 minutes
72        Tick::from_timetable_time(TimetableTime(60 * 30)).0, // 30 minutes
73        Tick::from_timetable_time(TimetableTime(60 * 60)).0, // 1 hour
74        Tick::from_timetable_time(TimetableTime(60 * 60 * 4)).0, // 4 hours
75        Tick::from_timetable_time(TimetableTime(60 * 60 * 24)).0, // 1 day
76    ];
77    let mut drawn: Vec<i64> = Vec::with_capacity(30);
78
79    // align the first tick to a spacing boundary that is <= visible start.
80    let first_visible_position = sizes
81        .iter()
82        .position(|s| *s as f64 / ticks_per_screen_unit * 1.5 > MIN_SCREEN_WIDTH)
83        .unwrap_or(0);
84    let visible = &sizes[first_visible_position..];
85    for (i, spacing) in visible.iter().enumerate().rev() {
86        let first = visible_ticks.start - visible_ticks.start.rem_euclid(*spacing) - spacing;
87        let mut tick = first;
88        let strength = (((*spacing as f64 / ticks_per_screen_unit * 1.5) - MIN_SCREEN_WIDTH)
89            / (MAX_SCREEN_WIDTH - MIN_SCREEN_WIDTH))
90            .clamp(0.0, 1.0);
91        if strength < 0.1 {
92            continue;
93        }
94        // TODO: make stroke depend on current theme
95        let mut current_stroke = Stroke {
96            width: 0.6,
97            color: Color32::GRAY,
98        };
99        if strength.is_finite() {
100            // strange bug here
101            current_stroke.color = current_stroke.color.gamma_multiply(strength as f32);
102        }
103        current_stroke.width = 0.5;
104        while tick <= visible_ticks.end {
105            tick += *spacing;
106            if drawn.contains(&tick) {
107                continue;
108            }
109            let mut x = ticks_to_screen_x(tick, screen_rect, ticks_per_screen_unit, tick_offset);
110            current_stroke.round_center_to_pixel(pixels_per_point, &mut x);
111            painter.vline(x, screen_rect.top()..=screen_rect.bottom(), current_stroke);
112            drawn.push(tick);
113            let time = Tick(tick).to_timetable_time();
114            let text = match i + first_visible_position {
115                0..=2 => time.to_hmsd().2.to_string(),
116                3..=8 => format!("{}:{:02}", time.to_hmsd().0, time.to_hmsd().1),
117                _ => time.to_string(),
118            };
119            let label = painter.layout_no_wrap(
120                text,
121                FontId::new(13.0, egui::FontFamily::Name("dia_pro".into())),
122                current_stroke.color,
123            );
124            painter.galley(
125                Pos2 {
126                    x: x - label.size().x / 2.0,
127                    y: screen_rect.top(),
128                },
129                label,
130                current_stroke.color,
131            );
132        }
133    }
