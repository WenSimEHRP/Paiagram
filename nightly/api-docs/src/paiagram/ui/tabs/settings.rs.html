1use crate::{
2    i18n::Language,
3    settings::{ProjectSettings, UserPreferences},
4};
5
6use super::Tab;
7use bevy::prelude::*;
8use egui::Ui;
9use egui_i18n::tr;
10use moonshine_core::prelude::MapEntities;
11use serde::{Deserialize, Serialize};
12use strum::IntoEnumIterator;
13
14#[derive(Serialize, Deserialize, Clone, Default, MapEntities, PartialEq)]
15pub struct SettingsTab;
16
17impl Tab for SettingsTab {
18    const NAME: &'static str = "Settings";
19    fn main_display(&mut self, world: &mut World, ui: &mut Ui) {
20        if let Err(e) = world.run_system_cached_with(show_settings, ui) {
21            bevy::log::error!("UI Error while displaying settings page: {}", e)
22        }
23    }
24    fn title(&self) -> egui::WidgetText {
25        tr!("tab-settings").into()
26    }
27}
28
29fn show_settings(
30    InMut(ui): InMut<Ui>,
31    mut preferences: ResMut<UserPreferences>,
32    mut settings: ResMut<ProjectSettings>,
33) {
34    ui.heading("Preferences");
35    ui.checkbox(&mut preferences.dark_mode, "Dark Mode");
36    egui::ComboBox::new(ui.id().with("settings box"), "Language").show_ui(ui, |ui| {
37        for lang in Language::iter() {
38            ui.selectable_value(&mut preferences.lang, lang, lang.name());
39        }
40    });
41    ui.heading("Project Settings");
42    ui.text_edit_multiline(&mut settings.remarks);
43}
