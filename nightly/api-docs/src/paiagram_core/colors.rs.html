24fn color_button(ui: &mut egui::Ui, color: Color32, open: bool) -> egui::Response {
25    let size = ui.spacing().interact_size;
26    let (rect, response) = ui.allocate_exact_size(size, egui::Sense::click());
27    response.widget_info(|| egui::WidgetInfo::new(egui::WidgetType::ColorButton));
28
29    if ui.is_rect_visible(rect) {
30        let visuals = if open {
31            &ui.visuals().widgets.open
32        } else {
33            ui.style().interact(&response)
34        };
35        let rect = rect.expand(visuals.expansion);
36
37        let stroke_width = 1.0;
38        show_color_at(ui.painter(), color, rect.shrink(stroke_width));
39
40        let corner_radius = visuals.corner_radius.at_most(2); // Can't do more rounding because the background grid doesn't do any rounding
41        ui.painter().rect_stroke(
42            rect,
43            corner_radius,
44            (stroke_width, visuals.bg_fill), // Using fill for stroke is intentional, because default style has no border
45            egui::StrokeKind::Inside,
46        );
47    }
48
49    response
50}

52impl egui::Widget for &mut DisplayColor {
53    fn ui(self, ui: &mut egui::Ui) -> egui::Response {
54        let is_dark = ui.visuals().dark_mode;
55        let button_res = color_button(ui, self.get(is_dark), false);
56
57        let current_predefined = match *self {
58            DisplayColor::Predefined(p) => Some(p),
59            DisplayColor::Custom(_) => None,
60        };
61
62        egui::Popup::menu(&button_res)
63            .close_behavior(egui::PopupCloseBehavior::CloseOnClickOutside)
64            .show(|ui| {
65                ui.horizontal(|ui| {
66                    ui.vertical(|ui| {
67                        ui.label("Predefined");
68                        ui.set_max_width(200.0);
69                        ui.horizontal_wrapped(|ui| {
70                            ui.style_mut().spacing.item_spacing = egui::Vec2::splat(4.0);
71                            for predefined in PredefinedColor::iter() {
72                                let color = predefined.get(is_dark);
73                                let is_selected = current_predefined == Some(predefined);
74                                let button = egui::Button::new("")
75                                    .fill(color)
76                                    .min_size(egui::vec2(24.0, 24.0))
77                                    .stroke(if is_selected {
78                                        ui.visuals().selection.stroke
79                                    } else {
80                                        ui.visuals().widgets.inactive.bg_stroke
81                                    });
82
83                                if ui.add(button).clicked() {
84                                    *self = DisplayColor::Predefined(predefined);
85                                }
86                            }
87                        });
88                    });
89                    ui.separator();
90                    ui.vertical(|ui| {
91                        ui.label("Custom");
92                        let mut custom_color = match *self {
93                            DisplayColor::Custom(c) => c,
94                            DisplayColor::Predefined(p) => p.get(is_dark),
95                        };
96                        if color_picker_color32(ui, &mut custom_color, Alpha::Opaque) {
97                            *self = DisplayColor::Custom(custom_color);
98                        }
99                    });
100                })
101            });
102        button_res
103    }
104}

225pub const fn translate_srgba_to_color32(c: Srgba) -> Color32 {
226    Color32::from_rgba_unmultiplied_const(
227        (c.red * 256.0) as u8,
228        (c.green * 256.0) as u8,
229        (c.blue * 256.0) as u8,
230        (c.alpha * 256.0) as u8,
231    )
232}

234/// Give the text colour that is readable given some background colour.
235fn readable_text_color(color: Color32) -> Color32 {
236    let [r, g, b, _] = color.to_array();
237    let luma = (0.299 * r as f32 + 0.587 * g as f32 + 0.114 * b as f32) / 255.0;
238    if luma > 0.5 {
239        Color32::BLACK
240    } else {
241        Color32::WHITE
242    }
