37pub fn parse_oud2_to_ast(file: &str) -> Result<Structure<'_>, pest::error::Error<Rule>> {
38    let oud2 = OUD2Parser::parse(Rule::file, file)?
39        .next()
40        .unwrap()
41        .into_inner()
42        .next()
43        .unwrap();
44    use pest::iterators::Pair;
45    fn parse_struct(pair: Pair<Rule>) -> Structure {
46        match pair.as_rule() {
47            Rule::r#struct => {
48                let mut inner = pair.into_inner();
49                let name = inner.next().unwrap().as_str();
50                let mut fields = Vec::new();
51                for field_pair in inner {
52                    let field_struct = parse_struct(field_pair);
53                    fields.push(field_struct);
54                }
55                Structure::Struct(name, fields)
56            }
57            Rule::wrapper => {
58                let inner = pair.into_inner();
59                let name = "file";
60                let mut fields = Vec::new();
61                for field_pair in inner {
62                    let field_struct = parse_struct(field_pair);
63                    fields.push(field_struct);
64                }
65                Structure::Struct(name, fields)
66            }
67            Rule::kvpair => {
68                let mut inner = pair.into_inner();
69                let key = inner.next().unwrap().as_str();
70                let val = inner.next().unwrap();
71                let val = match val.as_rule() {
72                    Rule::value => Value::Single(val.as_str()),
73                    Rule::list => {
74                        let list_vals = val.into_inner().map(|v| v.as_str()).collect();
75                        Value::List(list_vals)
76                    }
77                    _ => unreachable!(),
78                };
79                Structure::Pair(key, val)
80            }
81            _ => unreachable!(),
82        }
83    }
84    Ok(parse_struct(oud2))
85}

87pub fn load_oud(
88    msg: On<super::LoadOuDia>,
89    mut commands: Commands,
90    mut graph: ResMut<Graph>,
91    class_resource: Res<ClassResource>,
92) {
93    let str = match &msg.content {
94        OuDiaContentType::OuDiaSecond(s) => std::borrow::Cow::Borrowed(s.as_str()),
95        OuDiaContentType::OuDia(d) => {
96            use encoding_rs::SHIFT_JIS;
97            let (s, _, _) = SHIFT_JIS.decode(d);
98            s
99        }
100    };
101    info!("Loading OUD/OUD2 data...");
102    let ast = parse_oud2_to_ast(&str).expect("Failed to parse OUD/OUD2 file");
103    let root = parse_ast(&ast).expect("Failed to convert OUD/OUD2 AST to internal representation");
104    let mut station_map: HashMap<String, Instance<StationComponent>> = HashMap::new();
105    for line in root.lines {
106        let mut stations: Vec<Option<Instance<StationComponent>>> = vec![None; line.stations.len()];
107        let mut break_flags: Vec<bool> = Vec::with_capacity(line.stations.len());
108        for (i, station) in line.stations.iter().enumerate() {
109            let referenced = station
110                .branch_index
111                .or(station.loop_index)
112                .and_then(|idx| stations.get(idx).and_then(|e| *e));
113            // a bit slower but standardized
114            let station_entity = referenced.unwrap_or_else(|| {
115                super::make_station(&station.name, &mut station_map, &mut graph, &mut commands)
116            });
117            stations[i] = Some(station_entity);
118            break_flags.push(station.break_interval);
119        }
120
121        let station_instances: Vec<Instance<StationComponent>> =
122            stations.into_iter().map(|e| e.unwrap()).collect();
123        let class_instances: Vec<Entity> = line
124            .classes
125            .into_iter()
126            .map(|it| {
127                commands
128                    .spawn(ClassBundle {
129                        class: ClassComponent::default(),
130                        name: Name::new(it.name),
131                        stroke: DisplayedStroke {
132                            color: DisplayColor::Custom(it.color),
133                            width: 1.0,
134                        },
135                    })
136                    .id()
137            })
138            .collect();
139
140        commands.spawn((
141            Name::new(line.name),
142            Route {
143                stops: station_instances.iter().map(|e| e.entity()).collect(),
144                lengths: vec![10.0; station_instances.len()],
145            },
146        ));
147
148        for i in 0..station_instances.len().saturating_sub(1) {
149            if break_flags[i] {
150                continue;
151            }
152            super::add_interval_pair(
153                &mut graph,
154                &mut commands,
155                station_instances[i].entity(),
156                station_instances[i + 1].entity(),
157                Distance::from_m(1000),
158            );
159        }
160
161        // TODO: find a method to support multiple diagrams
162        for diagram in line.diagrams.into_iter().take(1) {
163            for train in diagram.trains {
164                let trip_class = train
165                    .class_index
166                    .map_or(class_resource.default_class, |idx| class_instances[idx]);
167                commands
168                    .spawn(TripBundle::new(&train.name, TripClass(trip_class.entity())))
169                    .with_children(|bundle| {
170                        for (stop, mut times) in &train
171                            .times
172                            .into_iter()
173                            .enumerate()
174                            .filter_map(|(i, time)| {
175                                let time = time?;
176                                if matches!(time.passing_mode, PassingMode::NoOperation) {
177                                    return None;
178                                }
179                                let station_index = match train.direction {
180                                    Direction::Down => i,
181                                    Direction::Up => station_instances.len() - 1 - i,
182                                };
183                                let stop = station_instances[station_index];
184                                Some((stop, time))
185                            })
186                            .chunk_by(|(s, _t)| *s)
187                        {
188                            let (_, first_time) = times.next().unwrap();
189                            let last_time = times.last().map(|(_, t)| t).unwrap_or(first_time);
190                            let arrival = if matches!(first_time.passing_mode, PassingMode::Pass) {
191                                None
192                            } else {
193                                Some(
194                                    first_time
195                                        .arrival
196                                        .map_or(TravelMode::Flexible, |t| TravelMode::At(t)),
197                                )
198                            };
199                            let departure = last_time
200                                .departure
201                                .map_or(TravelMode::Flexible, |t| TravelMode::At(t));
202                            bundle.spawn(EntryBundle::new(arrival, departure, stop.entity()));
203                        }
204                    });
205            }
206        }
207    }
208}

278pub fn parse_ast(ast: &Structure) -> Result<Root> {
279    let Struct(_, v) = ast else {
280        return Err(anyhow!("Expected root structure"));
281    };
282    let mut version = Option::None;
283    let mut lines = Vec::new();
284    let mut unnamed_line_counter = 0;
285    for field in v {
286        match field {
287            Struct(k, v) if *k == "Rosen" => {
288                lines.push(parse_line_meta(v, &mut unnamed_line_counter)?);
289            }
290            Pair(k, Single(v)) if *k == "FileType" => {
291                version = Some(v.to_string());
292            }
293            _ => {}
294        }
295    }
296    Ok(Root {
297        version: version.ok_or(anyhow!("File does not have a version"))?,
298        lines,
299    })
300}
301fn parse_line_meta(fields: &[Structure], unnamed_line_counter: &mut usize) -> Result<LineMeta> {
302    let mut name: Option<String> = None;
303    let mut stations = Vec::new();
304    let mut diagrams = Vec::new();
305    let mut classes = Vec::new();
306    let mut unnamed_station_counter = 0;
307    let mut unnamed_diagram_counter = 0;
308    let mut unnamed_train_counter = 0;
309    for field in fields {
310        match field {
311            Pair(k, Single(v)) if *k == "Rosenmei" => {
312                name = Some(v.to_string());
313            }
314            Struct(k, v) if *k == "Eki" => {
315                stations.push(parse_station(v, &mut unnamed_station_counter)?);
316            }
317            Struct(k, v) if *k == "Dia" => {
318                diagrams.push(parse_diagram(
319                    v,
320                    &mut unnamed_diagram_counter,
321                    &mut unnamed_train_counter,
322                )?);
323            }
324            Struct(k, v) if *k == "Ressyasyubetsu" => classes.push(parse_class(v)),
325            _ => {}
326        }
327    }
328    Ok(LineMeta {
329        name: name.unwrap_or_else(|| {
330            *unnamed_line_counter += 1;
331            let name = tr!("oud2-unnamed-line", {
332                number: unnamed_line_counter.to_string()
333            });
334            name
335        }),
336        stations,
337        diagrams,
338        classes,
339    })
340}

342fn parse_station(fields: &[Structure], unnamed_station_counter: &mut usize) -> Result<Station> {
343    let mut name: Option<String> = None;
344    let mut branch_index: Option<usize> = None;
345    let mut loop_index: Option<usize> = None;
346    let mut kudari_display = false;
347    let mut nobori_display = false;
348    for field in fields {
349        match field {
350            Pair(k, Single(v)) if *k == "Ekimei" => {
351                name = Some(v.to_string());
352            }
353            // The "brunch" here is intended - it is spelling mistake in the original software
354            Pair(k, Single(v)) if *k == "BrunchCoreEkiIndex" => {
355                branch_index = Some(
356                    v.parse::<usize>()
357                        .map_err(|e| anyhow!("Failed to parse branch index: {}", e))?,
358                );
359            }
360            Pair(k, Single(v)) if *k == "LoopOriginEkiIndex" => {
361                loop_index = Some(
362                    v.parse::<usize>()
363                        .map_err(|e| anyhow!("Failed to parse loop index: {}", e))?,
364                );
365            }
366            Pair(k, List(v)) if *k == "JikokuhyouJikokuDisplayKudari" => {
367                kudari_display = v.len() == 2 && [v[0], v[1]] == ["1", "0"];
368            }
369            Pair(k, List(v)) if *k == "JikokuhyouJikokuDisplayNobori" => {
370                nobori_display = v.len() == 2 && [v[0], v[1]] == ["0", "1"];
371            }
372            _ => {}
373        }
374    }
375    let break_interval = kudari_display && nobori_display;
376    Ok(Station {
377        name: name.unwrap_or_else(|| {
378            *unnamed_station_counter += 1;
379            let name = tr!("oud2-unnamed-station", {
380                number: unnamed_station_counter.to_string()
381            });
382            name
383        }),
384        branch_index,
385        loop_index,
386        break_interval,
387    })
388}

390fn parse_diagram(
391    fields: &[Structure],
392    unnamed_diagram_counter: &mut usize,
393    unnamed_train_counter: &mut usize,
394) -> Result<Diagram> {
395    let mut name: Option<String> = None;
396    let mut trains: Vec<Train> = Vec::new();
397    for field in fields {
398        match field {
399            Pair(k, Single(v)) if *k == "DiaName" => {
400                name = Some(v.to_string());
401            }
402            Struct(k, v) if *k == "Kudari" => {
403                trains.extend(parse_trains(Direction::Down, v, unnamed_train_counter)?);
404            }
405            Struct(k, v) if *k == "Nobori" => {
406                trains.extend(parse_trains(Direction::Up, v, unnamed_train_counter)?);
407            }
408            _ => {}
409        }
410    }
411    Ok(Diagram {
412        name: name.unwrap_or_else(|| {
413            *unnamed_diagram_counter += 1;
414            let name = tr!("oud2-unnamed-diagram", {
415                number: unnamed_diagram_counter.to_string()
416            });
417            name
418        }),
419        trains,
420    })
421}

423fn parse_trains(
424    direction: Direction,
425    fields: &[Structure],
426    unnamed_train_counter: &mut usize,
427) -> Result<Vec<Train>> {
428    fn parse_time(str: &str) -> Result<Option<TimetableEntry>> {
429        let mut entry = TimetableEntry {
430            passing_mode: PassingMode::NoOperation,
431            arrival: None,
432            departure: None,
433            track: None,
434        };
435        if str.is_empty() {
436            return Ok(None);
437        }
438        let parts = OUD2Parser::parse(Rule::timetable_entry, str)
439            .map_err(|e| anyhow!(e))?
440            .next()
441            .ok_or(anyhow!("Unexpected error while unwrapping"))?;
442        for field in parts.into_inner() {
443            match field.as_rule() {
444                Rule::service_mode => match field.as_str() {
445                    "1" => entry.passing_mode = PassingMode::Stop,
446                    "2" => entry.passing_mode = PassingMode::Pass,
447                    _ => entry.passing_mode = PassingMode::NoOperation,
448                },
449                Rule::arrival => entry.arrival = TimetableTime::from_oud2_str(field.as_str()),
450                Rule::departure => entry.departure = TimetableTime::from_oud2_str(field.as_str()),
451                Rule::track => {
452                    // TODO
453                }
454                _ => {}
455            }
456        }
457        Ok(Some(entry))
458    }
459    let mut parse_trains = |fields: &[Structure]| -> Result<Train> {
460        let mut name: Option<String> = None;
461        let mut entries: Vec<Option<TimetableEntry>> = Vec::new();
462        let mut class_index: Option<usize> = None;
463        for field in fields {
464            match field {
465                Pair(k, Single(v)) if *k == "Ressyabangou" && !v.trim().is_empty() => {
466                    name = Some(v.to_string());
467                }
468                Pair(k, v) if *k == "EkiJikoku" => {
469                    let times = match v {
470                        Single(s) => &vec![*s],
471                        List(l) => l,
472                    };
473                    for time in times {
474                        entries.push(parse_time(time)?);
475                    }
476                }
477                Pair(k, Single(v)) if *k == "Syubetsu" => {
478                    class_index = Some(v.parse::<usize>().map_err(|e| anyhow!("{:?}", e))?)
479                }
480                _ => {}
481            }
482        }
483        let time_iter = entries.iter_mut().flat_map(|t| {
484            std::iter::once(t).flatten().flat_map(|t| {
485                std::iter::once(&mut t.arrival)
486                    .flatten()
487                    .chain(std::iter::once(&mut t.departure).flatten())
488            })
489        });
490        super::normalize_times(time_iter);
491        Ok(Train {
492            direction,
493            name: name.unwrap_or_else(|| {
494                *unnamed_train_counter += 1;
495                let name = tr!("oud2-unnamed-train", {
496                    number: unnamed_train_counter.to_string()
497                });
498                name
499            }),
500            times: entries,
501            class_index,
502        })
503    };
504    let mut trains = Vec::new();
505    for field in fields {
506        match field {
507            Struct(k, v) if *k == "Ressya" => {
508                trains.push(parse_trains(v)?);
509            }
510            _ => {}
511        }
512    }
513    Ok(trains)
514}

516fn parse_class(fields: &[Structure]) -> TrainClass {
517    let mut name: Option<String> = None;
518    let mut color: Option<egui::Color32> = None;
519    for field in fields {
520        match field {
521            Pair(k, Single(v)) if *k == "Syubetsumei" => {
522                name = Some(v.to_string());
523            }
524            Pair(k, Single(v)) if *k == "DiagramSenColor" => {
525                // AARRGGBB
526                let (r, g, b) = (
527                    u8::from_str_radix(&v[2..=3], 16).unwrap(),
528                    u8::from_str_radix(&v[4..=5], 16).unwrap(),
529                    u8::from_str_radix(&v[6..=7], 16).unwrap(),
530                );
531                color = Some(egui::Color32::from_rgb(r, g, b))
532            }
533            _ => {}
534        }
535    }
536    TrainClass {
537        name: name.unwrap(),
538        color: color.unwrap(),
539    }
