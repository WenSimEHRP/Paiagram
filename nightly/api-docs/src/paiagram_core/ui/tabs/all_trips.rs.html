37impl super::Tab for AllTripsTab {
38    const NAME: &'static str = "All Trips";
39    fn scroll_bars(&self) -> [bool; 2] {
40        [false; 2]
41    }
42    fn edit_display(&mut self, world: &mut World, ui: &mut Ui) {
43        if ui.button("Sort entries").clicked() {
44            world.trigger(SortRouteByDirectionTrips {
45                entity: self.route_entity,
46            });
47        }
48    }
49    fn main_display(&mut self, world: &mut bevy::ecs::world::World, ui: &mut egui::Ui) {
50        let route = world.get::<Route>(self.route_entity).unwrap();
51        let by_direction = world
52            .get::<RouteByDirectionTrips>(self.route_entity)
53            .expect("Route should have RouteByDirectionTrips");
54        let downward_entities = by_direction.downward.clone();
55        // use a table
56        let table = egui_table::Table::new()
57            .id_salt(self.route_entity)
58            .num_rows(route.stops.len() as u64)
59            .num_sticky_cols(2);
60        world
61            .run_system_cached_with(
62                display_table,
63                (table, ui, self.route_entity, downward_entities.as_slice()),
64            )
65            .unwrap();
66    }
67}

69struct AllTripsDisplayer<'w> {
70    route: &'w Route,
71    route_display_modes: &'w mut RouteDisplayModes,
72    names: &'w [&'w str],
73    available_trips: &'w [Entity],
74    column_offset: usize,
75    trips: Vec<(TripQueryItem<'w, 'w>, Vec<EntryDisplayMode<'w>>)>,

90impl<'w> TableDelegate for AllTripsDisplayer<'w> {
91    fn prepare(&mut self, info: &egui_table::PrefetchInfo) {
92        self.trips.clear();
93
94        let visible_trip_cols_start = info.visible_columns.start.max(2);
95        let visible_trip_cols_end = info.visible_columns.end;
96
97        if visible_trip_cols_start >= visible_trip_cols_end {
98            return;
99        }
100
101        let trip_start = visible_trip_cols_start - 2;
102        let trip_end = visible_trip_cols_end - 2;
103        self.column_offset = trip_start;
104
105        let trips_iter = self
106            .trip_q
107            .iter_many(self.available_trips[trip_start..trip_end].iter())
108            .map(|it| {
109                let mut v = Vec::with_capacity(self.route.stops.len());
110                v.resize_with(self.route.stops.len(), || EntryDisplayMode::Skipped);
111                let schedule_it = self.entry_q.iter_many(it.schedule.iter());
112                let mut next_abs_idx = 0;
113                let mut stations = self.route.stops.iter();
114                for it in schedule_it {
115                    let station_entity = self
116                        .parent_station_or_station
117                        .get(it.stop())
118                        .unwrap()
119                        .parent();
120                    // we reuse the same iterator here
121                    // the pointer would advance every time we use the .position() method
122                    if let Some(found_pos) = stations.position(|it| *it == station_entity) {
123                        let abs_idx = next_abs_idx + found_pos;
124                        v[abs_idx] = EntryDisplayMode::Some(it);
125                        next_abs_idx = abs_idx + 1;
126                    }
127                }
128                for i in v
129                    .iter_mut()
130                    .take_while(|it| matches!(it, EntryDisplayMode::Skipped))
131                {
132                    *i = EntryDisplayMode::NoOperation
133                }
134                let mut last_processed: Option<&mut _> = None;
135                for i in v
136                    .iter_mut()
137                    .rev()
138                    .take_while(|it| matches!(it, EntryDisplayMode::Skipped))
139                {
140                    *i = EntryDisplayMode::NoOperation;
141                    last_processed = Some(i);
142                }
143                if let Some(m) = last_processed {
144                    *m = EntryDisplayMode::Terminated;
145                }
146                (it, v)
147            });
148        self.trips.extend(trips_iter);
149    }
150    fn row_top_offset(&self, _ctx: &egui::Context, _table_id: egui::Id, row_nr: u64) -> f32 {
151        let offset_count: usize = self.route_display_modes[0..(row_nr as usize)]
152            .iter()
153            .map(|mode| mode.count())
154            .sum();
155
156        (offset_count as f32) * self.default_row_height()
157    }
158    fn default_row_height(&self) -> f32 {
159        16.0

161    fn header_cell_ui(&mut self, ui: &mut egui::Ui, cell: &egui_table::HeaderCellInfo) {
162        let dy = vec2(0.0, -ui.visuals().window_stroke.width);
163        ui.painter().line_segment(
164            [
165                ui.max_rect().left_bottom() + dy,
166                ui.max_rect().right_bottom() + dy,
167            ],
168            ui.visuals().window_stroke(),
169        );
170        if cell.group_index == 0 {
171            ui.label("Stations");
172            return;
173        }
174        let dx = vec2(-ui.visuals().window_stroke.width, 0.0);
175        ui.painter().line_segment(
176            [
177                ui.max_rect().right_top() + dx,
178                ui.max_rect().right_bottom() + dx,
179            ],
180            ui.visuals().window_stroke(),
181        );
182        if cell.group_index == 1 {
183            return;
184        }
185        let trip_index = cell.group_index - 2;
186        let trip_end = self.column_offset + self.trips.len();
187        if trip_index < self.column_offset || trip_index >= trip_end {
188            return;
189        }
190        let local_trip_index = trip_index - self.column_offset;
191        let (t, _) = &self.trips[local_trip_index];
192        ui.label(t.name.as_str());
193    }
194    fn cell_ui(&mut self, ui: &mut egui::Ui, cell: &egui_table::CellInfo) {
195        let row_nr = cell.row_nr as usize;
196        let display_mode = &self.route_display_modes[row_nr];
197        if cell.col_nr == 0 {
198            ui.allocate_ui_with_layout(
199                ui.available_size(),
200                Layout::left_to_right(egui::Align::Center),
201                |ui| {
202                    ui.label(self.names[row_nr]);
203                },
204            );
205            return;
206        }
207        let dx = vec2(-ui.visuals().window_stroke.width, 0.0);
208        ui.painter().line_segment(
209            [
210                ui.max_rect().right_top() + dx,
211                ui.max_rect().right_bottom() + dx,
212            ],
213            ui.visuals().window_stroke(),
214        );
215        if cell.col_nr == 1 {
216            let prev_arr = row_nr
217                .checked_sub(1)
218                .map_or(false, |idx| self.route_display_modes[idx].arrival);
219            let prev_dep = !display_mode.arrival
220                && row_nr
221                    .checked_sub(1)
222                    .map_or(false, |idx| self.route_display_modes[idx].departure);
223            let m = &mut self.route_display_modes[row_nr];
224            let show_edit_button = |m: &mut AllTripsDisplayMode, ui: &mut egui::Ui, s: &str| {
225                let res = ui.button(s);
226                egui::Popup::menu(&res).show(|ui| {
227                    ui.add_enabled(
228                        !m.arrival || m.departure,
229                        egui::Checkbox::new(&mut m.arrival, "Arrival"),
230                    );
231                    ui.add_enabled(
232                        !m.departure || m.arrival,
233                        egui::Checkbox::new(&mut m.departure, "Departure"),
234                    );
235                });
236            };
237            ui.vertical(|ui| {
238                if m.arrival {
239                    let s = if prev_arr { "〃" } else { "Ａ" };
240                    show_edit_button(m, ui, s);
241                }
242                if m.departure {
243                    let s = if prev_dep { "〃" } else { "Ｄ" };
244                    show_edit_button(m, ui, s);
245                }
246            });
247            return;
248        }
249        if display_mode.arrival {
250            if display_mode.departure {
251                ui.painter().line_segment(
252                    [ui.max_rect().left_center(), ui.max_rect().right_center()],
253                    ui.visuals().window_stroke(),

259                        ui.max_rect().left_bottom() + dy,
260                        ui.max_rect().right_bottom() + dy,
261                    ],
262                    ui.visuals().window_stroke(),
263                );
264            }
265        }
266        let trip_index = cell.col_nr - 2;
267        let trip_end = self.column_offset + self.trips.len();
268        if trip_index < self.column_offset || trip_index >= trip_end {
269            return;
270        }
271        let local_trip_index = trip_index - self.column_offset;
272        let (_, entries) = &self.trips[local_trip_index];
273        let entry = &entries[row_nr];
274        ui.vertical(|ui| {
275            if display_mode.arrival {
276                let font = FontId::new(15.0, egui::FontFamily::Name("dia_pro".into()));
277                let res = ui.put(
278                    Rect::from_min_size(ui.max_rect().left_top(), AllTripsDisplayer::cell_size()),
279                    |ui: &mut egui::Ui| match entry {
280                        EntryDisplayMode::Skipped => ui.button(RichText::new("║").font(font)),
281                        EntryDisplayMode::NoOperation => ui.button(
282                            RichText::new(
283                                if (cell.row_nr + 1) % 10 == 0 && display_mode.count() < 2 {
284                                    "┄"
285                                } else {
286                                    "‥"
287                                },
288                            )
289                            .font(font),
290                        ),
291                        EntryDisplayMode::Terminated => ui.button(RichText::new("▔").font(font)),
292                        EntryDisplayMode::Some(e) => match e.mode.arr {
293                            Some(TravelMode::At(t)) => {
294                                let mut new_t = t;
295                                ui.add(
296                                    egui::DragValue::new(&mut new_t)
297                                        .custom_formatter(|it, _| {
298                                            TimetableTime::from_f64(it).to_oud2_str(false)
299                                        })
300                                        .custom_parser(|s| {

373fn display_table(
374    (In(table), InMut(ui), In(route_entity), InRef(trips_to_display)): (
375        In<Table>,
376        InMut<egui::Ui>,
377        In<Entity>,
378        InRef<[Entity]>,
379    ),
380    mut route_q: Query<(&Route, &mut RouteDisplayModes)>,
381    trip_q: Query<TripQuery>,
382    entry_q: Query<EntryQuery>,
383    parent_station_or_station: Query<ParentStationOrStation>,
384    names: Query<&Name, With<Station>>,
385) {
386    let (route, mut route_display_modes) = route_q.get_mut(route_entity).unwrap();
387    let names: Vec<_> = names
388        .iter_many(route.stops.iter())
389        .map(|it| it.as_str())
390        .collect();
391    let mut displayer = AllTripsDisplayer {
392        route,
393        route_display_modes: &mut *route_display_modes,
394        names: &names,
395        trips: Vec::new(),
396        available_trips: trips_to_display,
397        column_offset: 0,
398        trip_q: &trip_q,
399        entry_q: &entry_q,
400        parent_station_or_station: &parent_station_or_station,
401    };
402    let dia_pro_style = egui::TextStyle::Name("dia_pro".into());
403    ui.style_mut().text_styles.insert(
404        dia_pro_style.clone(),
405        egui::FontId::new(15.0, egui::FontFamily::Name("dia_pro".into())),
406    );
407    ui.style_mut().drag_value_text_style = dia_pro_style;
408    ui.spacing_mut().interact_size = Vec2::ZERO;
409    ui.spacing_mut().button_padding = Vec2::ZERO;
410    ui.style_mut().visuals.button_frame = false;
411    table
412        .columns(
413            std::iter::once(Column::new(80.0).resizable(true))
414                .chain(std::iter::once(Column::new(20.0).resizable(false)))
415                .chain(
416                    (0..trips_to_display.len()).map(|_| {
417                        Column::new(AllTripsDisplayer::table_cell_width()).resizable(false)
418                    }),
419                )
420                .collect::<Vec<_>>(),
421        )
422        .show(ui, &mut displayer);
