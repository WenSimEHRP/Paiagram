91fn show_start(
92    InMut(ui): InMut<Ui>,
93    vehicles: Query<VehicleQuery>,
94    trips: Query<TripQuery>,
95    stations: Query<StationQuery>,
96    platforms: Query<PlatformQuery>,
97    intervals: Query<IntervalQuery>,
98) {
99    ui.heading("Paiagram");
100    egui::Grid::new("start info grid")
101        .num_columns(2)
102        .show(ui, |ui| {
103            ui.label("Amount of vehicles:");
104            ui.label(vehicles.count().to_string());
105            ui.end_row();
106            ui.label("Amount of trips:");
107            ui.label(trips.count().to_string());
108            ui.end_row();
109            ui.label("Amount of stations:");
110            ui.label(stations.count().to_string());
111            ui.end_row();
112            ui.label("Amount of platforms:");
113            ui.label(platforms.count().to_string());
114            ui.end_row();
115            ui.label("Amount of intervals:");
116            ui.label(intervals.count().to_string());
117        });
118}
