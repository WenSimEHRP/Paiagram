21#[derive(SystemParam)]
22pub struct RaptorTimetable<'w, 's> {
23    station_query: Query<'w, 's, StationQuery>,
24    platform_entry_query: Query<'w, 's, &'static PlatformEntries>,
25    parent_query: Query<'w, 's, &'static ChildOf>,
26    parent_station_or_station_query: Query<'w, 's, ParentStationOrStation>,
27    trip_query: Query<'w, 's, TripQuery>,
28    entry_query: Query<'w, 's, EntryQuery>,
29}
30
31impl RaptorTimetable<'_, '_> {
32    fn normalize_stop(&self, stop: Entity) -> Option<Entity> {
33        self.parent_station_or_station_query
34            .get(stop)
35            .ok()
36            .map(|it| it.parent())
37    }
38
39    fn route_stop_sequence(&self, route: Entity) -> Vec<Entity> {
40        let Ok(trip) = self.trip_query.get(route) else {
41            return Vec::new();
42        };
43
44        self.entry_query
45            .iter_many(trip.schedule.iter())
46            .filter_map(|entry| self.normalize_stop(entry.stop()))
47            .collect()
48    }
49}
50
51impl raptor::Timetable for RaptorTimetable<'_, '_> {
52    type Route = Entity; // in this case it is the same as trips
53    type Stop = Entity;
54    type Trip = Entity;
55
56    fn get_routes_serving_stop(&self, stop: Self::Stop) -> Vec<Self::Route> {
57        let Ok(station) = self.station_query.get(stop) else {
58            return Vec::new();
59        };
60
61        let mut routes = std::collections::BTreeSet::new();
62        for entry in station.passing_entries(&self.platform_entry_query) {
63            if let Ok(parent) = self.parent_query.get(entry) {
64                routes.insert(parent.parent());
65            }
66        }
67        routes.into_iter().collect()
68    }
69
70    fn get_arrival_time(&self, trip: Self::Trip, stop: Self::Stop) -> raptor::Tau {
71        let Ok(route_trip) = self.trip_query.get(trip) else {
72            return raptor::Tau::MAX;
73        };
74
75        self.entry_query
76            .iter_many(route_trip.schedule.iter())
77            .find_map(|it| {
78                let normalized_stop = self.normalize_stop(it.stop())?;
79                if normalized_stop == stop {
80                    return it.estimate.map(|estimate| estimate.arr.0 as usize);
81                }
82                None
83            })
84            .unwrap_or(raptor::Tau::MAX)
85    }
86
87    fn get_departure_time(&self, trip: Self::Trip, stop: Self::Stop) -> raptor::Tau {
88        let Ok(route_trip) = self.trip_query.get(trip) else {
89            return raptor::Tau::MAX;
90        };
91
92        self.entry_query
93            .iter_many(route_trip.schedule.iter())
94            .find_map(|it| {
95                let normalized_stop = self.normalize_stop(it.stop())?;
96                if normalized_stop == stop {
97                    return it.estimate.map(|estimate| estimate.dep.0 as usize);
98                }
99                None
100            })
101            .unwrap_or(raptor::Tau::MAX)
102    }
103
104    fn get_earlier_stop(
105        &self,
106        route: Self::Route,
107        left: Self::Stop,
108        right: Self::Stop,
109    ) -> Self::Stop {
110        let stops = self.route_stop_sequence(route);
111        let pos_l = stops.iter().position(|&s| s == left);
112        let pos_r = stops.iter().position(|&s| s == right);
113
114        match (pos_l, pos_r) {
115            (Some(_), None) => left,
116            (None, Some(_)) => right,
117            (Some(l), Some(r)) => {
118                if l < r {
119                    left
120                } else {
121                    right
122                }
123            }
124            (None, None) => right,
125        }
126    }
127
128    fn get_earliest_trip(
129        &self,
130        route: Self::Route,
131        at: raptor::Tau,
132        boarding_stop: Self::Stop,
133    ) -> Option<Self::Trip> {
134        let dep = self.get_departure_time(route, boarding_stop);
135        if dep >= at && dep != raptor::Tau::MAX {
136            Some(route)
137        } else {
138            None
139        }
140    }
141
142    fn get_footpaths_from(&self, _stop: Self::Stop) -> Vec<Self::Stop> {
143        Vec::new()
144    }
145
146    fn get_stops_after(&self, route: Self::Route, stop: Self::Stop) -> Vec<Self::Stop> {
147        self.route_stop_sequence(route)
148            .into_iter()
149            .skip_while(|&s| s != stop)
150            .collect()
151    }
152}
